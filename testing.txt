// ============================================================
// GLOBAL WRITE SEMAPHORE
// ============================================================

const MAX_CONCURRENT_WRITES = 20
let activeWrites = 0
const globalWriteQueue = []

const acquireWriteSlot = () => {
    if (activeWrites < MAX_CONCURRENT_WRITES) {
        activeWrites++
        return Promise.resolve()
    }

    return new Promise(resolve => {
        globalWriteQueue.push(resolve)
    })
}

const releaseWriteSlot = () => {
    activeWrites--

    if (globalWriteQueue.length > 0) {
        activeWrites++
        const next = globalWriteQueue.shift()
        next()
    }
}



// ============================================================
// FD POOL
// ============================================================

const fdPool = new Map()

const getFd = async (uploadId, storagePath) => {
    let entry = fdPool.get(uploadId)

    if (!entry) {
        console.log(`[FD OPEN] ${uploadId.slice(-6)} opening fd`)

        const handle = await fsPromises.open(storagePath, "r+")

        entry = {
            fd: handle.fd,
            handle,
            lastUsed: Date.now()
        }

        fdPool.set(uploadId, entry)
    }

    entry.lastUsed = Date.now()

    return entry.fd
}

export const releaseFd = async (uploadId) => {
    const entry = fdPool.get(uploadId)

    if (!entry) return

    console.log(`[FD CLOSE] ${uploadId.slice(-6)} closing fd`)

    try {
        await entry.handle.close()
    } catch (err) {
        console.error(`[FD CLOSE ERROR] ${err.message}`)
    }

    fdPool.delete(uploadId)
}



// cleanup unused fds
setInterval(async () => {
    const now = Date.now()

    for (const [uploadId, entry] of fdPool.entries()) {
        if (now - entry.lastUsed > 2 * 60 * 1000) {
            console.log(`[FD CLEANUP] ${uploadId.slice(-6)} idle fd removed`)

            try {
                await entry.handle.close()
            } catch {}

            fdPool.delete(uploadId)
        }
    }
}, 2 * 60 * 1000)

const writeQueues = new Map()

const getWriteQueue = (uploadId) => {
    if (!writeQueues.has(uploadId)) {
        writeQueues.set(uploadId, Promise.resolve())
    }

    return writeQueues.get(uploadId)
}


// IMPORTANT
// now allowing small concurrency per file
const PER_FILE_PARALLEL_WRITES = 3

const activeFileWrites = new Map()
const waitingFileWrites = new Map()

const acquireFileSlot = (uploadId) => {
    const active = activeFileWrites.get(uploadId) || 0

    if (active < PER_FILE_PARALLEL_WRITES) {
        activeFileWrites.set(uploadId, active + 1)
        return Promise.resolve()
    }

    return new Promise(resolve => {
        if (!waitingFileWrites.has(uploadId)) {
            waitingFileWrites.set(uploadId, [])
        }

        waitingFileWrites.get(uploadId).push(resolve)
    })
}

const releaseFileSlot = (uploadId) => {
    const active = activeFileWrites.get(uploadId) || 0

    activeFileWrites.set(uploadId, Math.max(0, active - 1))

    const queue = waitingFileWrites.get(uploadId)

    if (queue && queue.length > 0) {
        activeFileWrites.set(uploadId, (activeFileWrites.get(uploadId) || 0) + 1)

        const next = queue.shift()
        next()
    }
}



// ============================================================
// UPLOAD CACHE
// ============================================================

const uploadRecordCache = new Map()
const CACHE_TTL = 10 * 60 * 1000

const getCachedUpload = async (uploadId, owner) => {
    const cached = uploadRecordCache.get(uploadId)

    if (cached && Date.now() - cached.cachedAt < CACHE_TTL) {
        return cached.record
    }

    console.log(`[CACHE MISS] upload ${uploadId.slice(-6)}`)

    const record = await Upload.findOne({ uploadId, owner })

    if (record) {
        uploadRecordCache.set(uploadId, {
            record,
            cachedAt: Date.now()
        })
    }

    return record
}

export const clearUploadCache = (uploadId) => {
    uploadRecordCache.delete(uploadId)
}



// ============================================================
// CHUNK MEMORY CACHE
// ============================================================

const uploadedChunkCache = new Map()

const getChunkSet = async (uploadId) => {
    if (uploadedChunkCache.has(uploadId)) {
        return uploadedChunkCache.get(uploadId)
    }

    console.log(`[CHUNK CACHE LOAD] ${uploadId.slice(-6)} loading uploaded chunks`)

    const chunks = await Chunk.find(
        { uploadId },
        { chunkIndex: 1, _id: 0 }
    ).lean()

    const set = new Set(chunks.map(c => c.chunkIndex))

    uploadedChunkCache.set(uploadId, set)

    return set
}

export const clearChunkCache = (uploadId) => {
    uploadedChunkCache.delete(uploadId)
}



// ============================================================
// BULK INSERT BUFFER
// ============================================================

const chunkInsertBuffers = new Map()

const flushChunkBuffer = async (uploadId) => {
    const buffer = chunkInsertBuffers.get(uploadId)

    if (!buffer || buffer.length === 0) return

    chunkInsertBuffers.set(uploadId, [])

    try {
        await Chunk.insertMany(buffer, { ordered: false })

        console.log(`[CHUNK BULK INSERT] ${uploadId.slice(-6)} inserted ${buffer.length} chunks`)
    } catch (err) {
        console.error(`[CHUNK BULK ERROR] ${err.message}`)
    }
}

const addChunkInsert = async (uploadId, chunkIndex) => {
    if (!chunkInsertBuffers.has(uploadId)) {
        chunkInsertBuffers.set(uploadId, [])
    }

    const arr = chunkInsertBuffers.get(uploadId)

    arr.push({ uploadId, chunkIndex })

    if (arr.length >= 50) {
        await flushChunkBuffer(uploadId)
    }
}

export const streamChunkMiddleware = (req, res, next) => {
    const owner = req.user._id

    const uploadId = req.headers["x-upload-id"]
    const indexes = JSON.parse(req.headers["x-indexes"] || "[]")
    const starts = JSON.parse(req.headers["x-starts"] || "[]")

    if (!uploadId) {
        return res.status(400).json({
            success: false,
            message: "Missing x-upload-id"
        })
    }

    console.log(``)
    console.log(`=====================================================`)
    console.log(`[UPLOAD START] ${uploadId.slice(-6)}`)
    console.log(`=====================================================`)

    const bb = busboy({
        headers: req.headers,
        limits: {
            fileSize: 10 * 1024 * 1024
        }
    })

    const writePromises = []



    // ========================================================
    // FILE
    // ========================================================

    bb.on("file", (fieldname, stream) => {
        const chunkBuffers = []

        stream.on("data", data => {
            chunkBuffers.push(data)
        })

        const promise = new Promise((resolve, reject) => {
            stream.on("end", async () => {
                const startedAt = Date.now()

                try {
                    const chunkIndex = parseInt(fieldname.split("_")[1])

                    const position = starts[indexes.indexOf(chunkIndex)]

                    const buffer = Buffer.concat(chunkBuffers)

                    console.log(`[CHUNK RECEIVE] chunk=${chunkIndex} size=${buffer.length}`)

                    if (position === undefined || position === -1) {
                        console.log(`[SKIP] invalid position chunk=${chunkIndex}`)
                        return resolve({ skipped: true })
                    }

                    const record = await getCachedUpload(uploadId, owner)

                    if (!record) {
                        console.log(`[SKIP] upload missing chunk=${chunkIndex}`)
                        return resolve({ skipped: true })
                    }

                    const uploadedSet = await getChunkSet(uploadId)

                    if (uploadedSet.has(chunkIndex)) {
                        console.log(`[SKIP] already uploaded chunk=${chunkIndex}`)
                        return resolve({ skipped: true })
                    }

                    await acquireWriteSlot()
                    await acquireFileSlot(uploadId)

                    try {
                        const fd = await getFd(uploadId, record.storagePath)

                        console.log(
                            `[WRITE START] chunk=${chunkIndex} pos=${position} activeGlobal=${activeWrites}`
                        )

                        // =============================================
                        // DIRECT FS.WRITE
                        // =============================================

                        await new Promise((resolveWrite, rejectWrite) => {
                            fs.write(
                                fd,
                                buffer,
                                0,
                                buffer.length,
                                position,
                                (err) => {
                                    if (err) return rejectWrite(err)
                                    resolveWrite()
                                }
                            )
                        })

                        uploadedSet.add(chunkIndex)

                        await addChunkInsert(uploadId, chunkIndex)

                        console.log(
                            `[WRITE OK] chunk=${chunkIndex} took=${Date.now() - startedAt}ms`
                        )

                        resolve({
                            skipped: false,
                            chunkIndex
                        })
                    }
                    finally {
                        releaseWriteSlot()
                        releaseFileSlot(uploadId)
                    }
                }
                catch (err) {
                    console.error(`[WRITE ERROR] ${uploadId} ${err.message}`)
                    reject(err)
                }
            })

            stream.on("error", err => {
                console.error(`[STREAM ERROR] ${err.message}`)
                reject(err)
            })
        })

        writePromises.push(promise)
    })



    // ========================================================
    // FINISH
    // ========================================================

    bb.on("finish", async () => {
        try {
            console.log(`[BUSBOY FINISH] waiting for writes...`)

            await Promise.all(writePromises)

            await flushChunkBuffer(uploadId)

            console.log(`[ALL WRITES DONE] ${uploadId.slice(-6)}`)

            if (!res.writableEnded) {
                res.json({
                    success: true,
                    message: "Written"
                })
            }
        }
        catch (err) {
            console.error(`[FINAL ERROR] ${err.message}`)

            if (!res.writableEnded) {
                res.status(500).json({
                    success: false,
                    message: err.message
                })
            }
        }
    })



    // ========================================================
    // ERROR
    // ========================================================

    bb.on("error", err => {
        console.error(`[BUSBOY ERROR] ${err.message}`)

        if (!res.writableEnded) {
            res.status(500).json({
                success: false,
                message: err.message
            })
        }
    })



    // ========================================================
    // PIPE
    // ========================================================

    req.pipe(bb)
}

export const cleanupUploadResources = async (uploadId) => {
    console.log(`[CLEANUP] ${uploadId.slice(-6)}`)

    await flushChunkBuffer(uploadId)

    await releaseFd(uploadId)

    writeQueues.delete(uploadId)
    activeFileWrites.delete(uploadId)
    waitingFileWrites.delete(uploadId)

    clearUploadCache(uploadId)
    clearChunkCache(uploadId)

    chunkInsertBuffers.delete(uploadId)
}

import axiosApi from "../../utils/api";
import { isBlockedFile } from "../../utils/blockFileTypes.js";


const CHUNK_SIZE = 7 * 1024 * 1024
const SMALL_FILE_THRESHOLD = 1 * 1024 * 1024
const SMALL_BATCH_SIZE = 25
const CHUNK_BATCH_SIZE = 3
const MAX_CONCURRENT = 3

export function useUploadWorkers(refs, updateFile, onUploadComplete, updateSession) {
    const { sessionMapsRef, uploadQueuesRef, uploadStartedRef, abortControllersRef } = refs

    //  this is the helper function for read 4100 byts of data as base64 for security checking MIME 
    const getFileHeader = (file) => {
        return new Promise((resolve) => {
            const slice = file.slice(0, 4100)
            const reader = new FileReader()
            reader.onload = (e) => {
                const base64 = btoa(String.fromCharCode(...new Uint8Array(e.target.result)))
                resolve(base64)
            }
            reader.readAsArrayBuffer(slice)
        })
    }


    //  this function is used for uploading multiple small file in one http request here
    const uploadSmallBatch = async (sessionId, batch, parentId) => {
        // STEP 1 — handle placeholders separately
        batch
            .filter(f => f.file._isPlaceholder)
            .forEach(f => {
                updateFile(sessionId, f.filekey, { status: "done", progress: 100 }, f.status)
            })

        // STEP 2 — filter real files
        const realBatch = batch.filter(f => !f.file._isPlaceholder)
        if (realBatch.length === 0) return

        // STEP 3 — block dangerous files (frontend validation)
        const safeBatch = []
        for (const f of realBatch) {
            if (isBlockedFile(f.file.name)) {
                updateFile(sessionId, f.filekey, {
                    status: "blocked",
                    message: "File type not allowed"
                }, f.status)
                continue
            }
            safeBatch.push(f)
        }

        if (safeBatch.length === 0) return

        // STEP 4 — create abort controller for batch
        const controller = new AbortController()
        safeBatch.forEach(f => {
            abortControllersRef.current.set(f.filekey, controller)
        })

        const form = new FormData()
        form.append("parent", parentId ? String(parentId) : null)

        try {
            // STEP 5 — get file headers for backend security check
            const fileHeaders = await Promise.all(
                safeBatch.map(f => getFileHeader(f.file))
            )

            const metadata = safeBatch.map((f, i) => ({
                fileName: f.file.name,
                fingerprint: f.fingerprint,
                fileSize: f.file.size,
                fileType: f.file.type || "application/octet-stream",
                parentId: f.parentId,
                fileHeader: fileHeaders[i],
                replacesFileId: f.replacesFileId || null
            }))

            form.append("metadata", JSON.stringify(metadata))

            // STEP 6 — append files and mark uploading
            safeBatch.forEach((f, i) => {
                form.append(`file_${i}`, f.file)
                updateFile(sessionId, f.filekey, { status: "uploading" }, f.status)
            })

            // STEP 7 — API call
            const { data } = await axiosApi.post("/upload/small-batch", form, {
                signal: controller.signal
            })

            // STEP 8 — handle response
            safeBatch.forEach(f => {
                const result = data.results?.find(
                    r => r.fingerprint === f.fingerprint
                )

                if (result?.status === "blocked") {
                    updateFile(sessionId, f.filekey, {
                        status: "blocked",
                        message: result.message || "File not allowed"
                    }, f.status)
                } else {
                    // store uploadId back to filesMap for cancel cleanup
                    const filesMap = sessionMapsRef.current.get(sessionId)
                    if (filesMap?.has(f.filekey)) {
                        filesMap.set(f.filekey, {
                            ...filesMap.get(f.filekey),
                            uploadId: result?.uploadId
                        })
                    }

                    updateFile(sessionId, f.filekey, {
                        status: "done",
                        progress: 100,
                        id: result?.id || null // Store the DB _id
                    })
                    // onUploadComplete?.()
                    // if (f.isRetry) onUploadComplete?.()
                }
            })

        } catch (error) {
            if (error.name === "CanceledError") return
            console.error("[DEBUG] uploadSmallBatch error response:", error.response?.data);
            console.error("uploadSmallBatch error:", error)

            const isNetworkError = !error.response
            const isBlocked = error.response?.status === 403 || error.response?.status === 400
            const errorMessage = isNetworkError ? "No connection" : error.response?.data?.message || "Upload failed"

            // Treat 403 (Permission) and 400 (Trash) as "blocked" status
            safeBatch.forEach(f => {
                updateFile(sessionId, f.filekey, {
                    status: isBlocked ? "blocked" : "error",
                    message: errorMessage
                }, f.status)
            })

            // if network is down then stop the whole queue
            if (isNetworkError) {
                const queue = uploadQueuesRef.current.get(sessionId)
                if (queue) {
                    // mark all remaining files in queue as error
                    queue.forEach(fileObj => {
                        if (fileObj === "DONE" || !fileObj) return
                        updateFile(sessionId, fileObj.filekey, {
                            status: "error",
                            message: "No connection"
                        }, fileObj.status)
                    })
                    queue.length = 0
                    queue.push("DONE")
                }
            }
        }
    }



    // handle large files uploading 
    const uploadLargeFile = async (sessionId, fileObj, replaceMap) => {
        //  file select then update progress bar
        // if (fileObj.file._isPlaceholder) {
        //     updateFile(sessionId, filekey, { status: "uploading" })
        // }
        if (fileObj.file._isPlaceholder) {
            updateFile(sessionId, fileObj.filekey, {
                status: "done",
                progress: 100
            }, fileObj.status)
            return; // STOP upload completely
        }

        const { file, filekey, fingerprint, parentId } = fileObj;
        updateFile(sessionId, filekey, { status: "uploading" })

        // create abort controller for this file
        const controller = new AbortController()
        abortControllersRef.current.set(filekey, controller)

        try {

            //  here we are blocking the dangorus file extension 
            if (isBlockedFile(file.name)) {
                updateFile(sessionId, filekey, { status: "blocked", message: "File type is not allowed" })
                return
            }
            const fileHeader = await getFileHeader(file)

            let uploadId, alreadyUploaded

            //  first init data to backend for check status
            const initStart = performance.now()
            const { data: initData } = await axiosApi.post("/upload/init", {
                fileName: file.name,
                fingerprint,
                fileSize: file.size,
                fileType: file.type || "application/octet-stream",
                totalChunks: Math.ceil(file.size / CHUNK_SIZE),
                parent: parentId || null,
                fileHeader,
                replacesFileId: replaceMap?.[file.name] || null
            }, { signal: controller.signal })


            //  if the mime is block from backend 
            if (initData.blocked) {
                console.warn(`[DEBUG] init blocked: ${file.name} | reason=${initData.message}`);
                updateFile(sessionId, filekey, { status: "blocked", message: initData.message })
                return
            }
            console.log(`[INIT] ${(performance.now() - initStart).toFixed(1)}ms | uploadId=${initData.uploadId}`);

            //  if file already uploaded then already uploaded
            if (initData.status === "completed") {
                updateFile(sessionId, filekey, { status: "skipped", progress: 100 }, fileObj.status)
                return
            }


            uploadId = initData.uploadId

            // store uploadId in fileObj for cancle use
            const filesMap = sessionMapsRef.current.get(sessionId)
            if (filesMap?.has(filekey)) {
                filesMap.set(filekey, { ...filesMap.get(filekey), uploadId })
            }


            alreadyUploaded = new Set(initData.uploadedChunks || [])    // how many chunks are already uploaded
            const totalChunks = Math.ceil(file.size / CHUNK_SIZE)

            // here skipp the chunks that are already uploaded
            const chunks = []
            for (let i = 0; i < totalChunks; i++) {
                if (alreadyUploaded.has(i)) continue
                const start = i * CHUNK_SIZE;
                const end = Math.min(start + CHUNK_SIZE, file.size)
                chunks.push({ index: i, start, blob: file.slice(start, end) })
            }

            //  in one batch how many chunk we want
            const batches = []
            for (let i = 0; i < chunks.length; i += CHUNK_BATCH_SIZE) {
                batches.push(chunks.slice(i, i + CHUNK_BATCH_SIZE))
            }

            let uploadCount = alreadyUploaded.size


            //  internet speed tracking here
            let bytesInWindow = 0
            let windowStart = Date.now()
            const SPEED_UPDATE_INTERVAL = 1500     // here every 1.5 second speed update


            //  send batch to the backend /upload-chunk
            console.log(`[CHUNKS] ${file.name} | Total batches: ${batches.length} | Total chunks: ${chunks.length} | Already uploaded: ${alreadyUploaded.size}`)
            for (let i = 0; i < batches.length; i++) {
                const batch = batches[i]
                const batchBytes = batch.reduce((sum, c) => sum + c.blob.size, 0)


                const form = new FormData()
                batch.forEach(c => form.append(`chunk_${c.index}`, c.blob))


                const chunkStart = performance.now()
                console.log(`[BATCH START] ${file.name} | batch ${i + 1}/${batches.length} | chunks: [${batch.map(c => c.index)}] | sending...`)

                let retries = 0
                const MAX_RETRIES = 3
                while (true) {
                    try {
                        await axiosApi.post("/upload/upload-chunk", form, {
                            signal: controller.signal,
                            timeout: 300000,   // 5 miniutes
                            headers: {
                                "x-upload-id": uploadId,
                                "x-indexes": JSON.stringify(batch.map(c => c.index)),
                                "x-starts": JSON.stringify(batch.map(c => c.start))
                            }
                        })
                        console.log(`[BATCH OK] ${file.name} | batch ${i + 1}/${batches.length} | took ${(performance.now() - chunkStart).toFixed(0)}ms`)
                        break // Success!
                    } catch (err) {
                        if (err.name === "CanceledError") return
                        console.error(`[BATCH FAIL DETAIL]`, {
                            fileName: file.name,
                            batchIndex: i,
                            totalBatches: batches.length,
                            errorName: err.name,
                            errorCode: err.code,
                            errorMessage: err.message,
                            httpStatus: err.response?.status,
                            httpData: err.response?.data,
                            isTimeout: err.code === "ECONNABORTED",
                            isNetworkError: !err.response,
                            retriesExhausted: retries >= MAX_RETRIES
                        })

                        const isNetworkError = !err.response
                        if (isNetworkError && retries < MAX_RETRIES) {
                            retries++
                            console.warn(`[RETRY] ${file.name} | batch ${i + 1}/${batches.length} | retry ${retries}/${MAX_RETRIES} | err=${err.message}`)
                            await new Promise(r => setTimeout(r, 2000 * retries)) // Wait 2s, 4s, 6s
                            continue
                        }

                        console.error(`[BATCH FAIL] ${file.name} | batch ${i + 1}/${batches.length} | took ${(performance.now() - chunkStart).toFixed(0)}ms | err.name=${err.name} | err.code=${err.code} | status=${err.response?.status} | msg=${err.message}`)
                        console.error(`[UPLOAD ERROR] ${new Date().toLocaleTimeString()} | File: ${file.name} | Status: ${err.response?.status} | Msg: ${err.message}`);
                        updateFile(sessionId, filekey, { status: "error", message: "Upload failed - connection lost" })
                        throw err
                    }
                }

                // update the speed
                uploadCount += batch.length;
                bytesInWindow += batchBytes


                const now = Date.now()
                const elapsed = now - windowStart

                const pct = parseFloat(((uploadCount / totalChunks) * 100).toFixed(1))

                if (elapsed >= SPEED_UPDATE_INTERVAL) {
                    const speedMBps = (bytesInWindow / (1024 * 1024)) / (elapsed / 1000)
                    updateFile(sessionId, filekey, {
                        progress: pct,
                        speed: speedMBps
                    })

                    // reset window
                    bytesInWindow = 0;
                    windowStart = now
                } else {
                    updateFile(sessionId, filekey, { progress: pct })
                }


            }

            console.log(`[ALL BATCHES DONE] ${file.name} | All ${batches.length} batches finished | calling /upload/complete...`)

            console.log(`[COMPLETE] calling complete for: ${file.name}, uploadId: ${uploadId}`)

            const { data: completeData } = await axiosApi.post("/upload/complete", { uploadId }, { signal: controller.signal })
            updateFile(sessionId, filekey, {
                status: "done",
                progress: 100,
                speed: null,
                id: completeData?.id || null // Store the DB _id
            }, fileObj.status)

            console.log(`[COMPLETE RESPONSE] ${file.name}: ${JSON.stringify(completeData)}`)

            // if (fileObj.isRetry) onUploadComplete?.()


        } catch (error) {
            console.error(`[LARGEFILE CATCH] ${file.name} | error.name=${error.name} | error.code=${error.code} | error.message=${error.message}`)
            if (error.name === "CanceledError") return
            console.error("[DEBUG] uploadLargeFile error response:", error.response?.data);
            console.error("uploadLargeFile error:", error)

            console.error(`[LARGEFILE CATCH DETAIL]`, {
                fileName: file.name,
                errorName: error.name,
                errorCode: error.code,
                errorMessage: error.message,
                httpStatus: error.response?.status,
                httpData: error.response?.data,
                isTimeout: error.code === "ECONNABORTED",
                isNetworkError: !error.response
            })

            // check if already marked as error from chunk failure
            const currentFile = sessionMapsRef.current.get(sessionId)?.get(filekey)
            if (currentFile?.status === "error") {
                console.warn(`[LARGEFILE CATCH] ${file.name} | Already marked as error, skipping`)
                return  // ← already handled, don't overwrite
            }

            const errorMessage = error.response?.data?.message || "Upload failed"
            const isBlocked = error.response?.status === 403 || error.response?.status === 400
            const isNetworkError = !error.response

            console.error(`[LARGEFILE CATCH] ${file.name} | Setting status=${isBlocked ? 'blocked' : 'error'} | msg=${errorMessage}`)
            updateFile(sessionId, filekey, {
                status: isBlocked ? "blocked" : "error",
                message: errorMessage
            }, fileObj.status)

            // if network is down then stop the whole queue (same as small file logic)
            if (isNetworkError) {
                console.error(`[LARGEFILE CATCH] ${file.name} | Network error — stopping entire queue`)
                const queue = uploadQueuesRef.current.get(sessionId)
                if (queue) {
                    queue.forEach(queuedFile => {
                        if (queuedFile === "DONE" || !queuedFile) return
                        updateFile(sessionId, queuedFile.filekey, {
                            status: "error",
                            message: "No connection"
                        }, queuedFile.status)
                    })
                    queue.length = 0
                    queue.push("DONE")
                }

            while (true) {

                if (fileObj === "DONE") {

                    queue.unshift("DONE")
                    break
                }

                } else {

                    await uploadLargeFile(sessionId, fileObj)
                    console.log(`[WORKER-${workerIndex}] large file done: ${fileObj.file.name} | took ${(performance.now() - workerFileStart).toFixed(0)}ms`)

            // add DONE back at end

        if (!filesMap) return

        // 1. Get only the files that actually failed
        const errorFiles = [];
        filesMap.forEach((fileObj) => {
            if (fileObj.status === "error") {
                errorFiles.push(fileObj)
            }
        })
        console.log(`[DEBUG] retryFolder: Found ${errorFiles.length} files with status 'error'`);
        if (errorFiles.length === 0) return

        // 2. Reset all error files to "waiting"
        errorFiles.forEach(fileObj => {
            // Capture the old "error" status so updateFile knows to subtract from it
            const oldStatus = fileObj.status;
            const oldProgress = fileObj.progress;

            const updated = {
                ...fileObj,
                status: "waiting",
                progress: 0,
                speed: null,
                isRetry: true

        queue.push("DONE");

        // 5. RESTART WORKERS
        // Clear the started flag so startUploadWorkers actually runs
        uploadStartedRef.current.set(sessionId, false);

//  here this file start uplaodihn all files and folder here 
import axiosApi from "../../utils/api";
import { isBlockedFile } from "../../utils/blockFileTypes.js";


const CHUNK_SIZE = 7 * 1024 * 1024
const SMALL_FILE_THRESHOLD = 1 * 1024 * 1024
const SMALL_BATCH_SIZE = 25
const CHUNK_BATCH_SIZE = 3
const MAX_CONCURRENT = 3

export function useUploadWorkers(refs, updateFile, onUploadComplete, updateSession) {
    const { sessionMapsRef, uploadQueuesRef, uploadStartedRef, abortControllersRef } = refs

    //  this is the helper function for read 4100 byts of data as base64 for security checking MIME 
    const getFileHeader = (file) => {
        return new Promise((resolve) => {
            const slice = file.slice(0, 4100)
            const reader = new FileReader()
            reader.onload = (e) => {
                const base64 = btoa(String.fromCharCode(...new Uint8Array(e.target.result)))
                resolve(base64)
            }
            reader.readAsArrayBuffer(slice)
        })
    }


    //  this function is used for uploading multiple small file in one http request here
    const uploadSmallBatch = async (sessionId, batch, parentId) => {
        // STEP 1 — handle placeholders separately
        batch
            .filter(f => f.file._isPlaceholder)
            .forEach(f => {
                updateFile(sessionId, f.filekey, { status: "done", progress: 100 }, f.status)
            })

        // STEP 2 — filter real files
        const realBatch = batch.filter(f => !f.file._isPlaceholder)
        if (realBatch.length === 0) return

        // STEP 3 — block dangerous files (frontend validation)
        const safeBatch = []
        for (const f of realBatch) {
            if (isBlockedFile(f.file.name)) {
                updateFile(sessionId, f.filekey, {
                    status: "blocked",
                    message: "File type not allowed"
                }, f.status)
                continue
            }
            safeBatch.push(f)
        }

        if (safeBatch.length === 0) return

        // STEP 4 — create abort controller for batch
        const controller = new AbortController()
        safeBatch.forEach(f => {
            abortControllersRef.current.set(f.filekey, controller)
        })

        const form = new FormData()
        form.append("parent", parentId ? String(parentId) : null)

        try {
            // STEP 5 — get file headers for backend security check
            const fileHeaders = await Promise.all(
                safeBatch.map(f => getFileHeader(f.file))
            )

            const metadata = safeBatch.map((f, i) => ({
                fileName: f.file.name,
                fingerprint: f.fingerprint,
                fileSize: f.file.size,
                fileType: f.file.type || "application/octet-stream",
                parentId: f.parentId,
                fileHeader: fileHeaders[i],
                replacesFileId: f.replacesFileId || null
            }))

            form.append("metadata", JSON.stringify(metadata))

            // STEP 6 — append files and mark uploading
            safeBatch.forEach((f, i) => {
                form.append(`file_${i}`, f.file)
                updateFile(sessionId, f.filekey, { status: "uploading" }, f.status)
            })

            // STEP 7 — API call
            const { data } = await axiosApi.post("/upload/small-batch", form, {
                signal: controller.signal
            })

            // STEP 8 — handle response
            safeBatch.forEach(f => {
                const result = data.results?.find(
                    r => r.fingerprint === f.fingerprint
                )

                if (result?.status === "blocked") {
                    updateFile(sessionId, f.filekey, {
                        status: "blocked",
                        message: result.message || "File not allowed"
                    }, f.status)
                } else {
                    // store uploadId back to filesMap for cancel cleanup
                    const filesMap = sessionMapsRef.current.get(sessionId)
                    if (filesMap?.has(f.filekey)) {
                        filesMap.set(f.filekey, {
                            ...filesMap.get(f.filekey),
                            uploadId: result?.uploadId
                        })
                    }

                    updateFile(sessionId, f.filekey, {
                        status: "done",
                        progress: 100,
                        id: result?.id || null // Store the DB _id
                    })
                    // onUploadComplete?.()
                    // if (f.isRetry) onUploadComplete?.()
                }
            })

        } catch (error) {
            if (error.name === "CanceledError") return
            console.error("[DEBUG] uploadSmallBatch error response:", error.response?.data);
            console.error("uploadSmallBatch error:", error)

            const isNetworkError = !error.response
            const isBlocked = error.response?.status === 403 || error.response?.status === 400
            const errorMessage = isNetworkError ? "No connection" : error.response?.data?.message || "Upload failed"

            // Treat 403 (Permission) and 400 (Trash) as "blocked" status
            safeBatch.forEach(f => {
                updateFile(sessionId, f.filekey, {
                    status: isBlocked ? "blocked" : "error",
                    message: errorMessage
                }, f.status)
            })

            // if network is down then stop the whole queue
            if (isNetworkError) {
                const queue = uploadQueuesRef.current.get(sessionId)
                if (queue) {
                    // mark all remaining files in queue as error
                    queue.forEach(fileObj => {
                        if (fileObj === "DONE" || !fileObj) return
                        updateFile(sessionId, fileObj.filekey, {
                            status: "error",
                            message: "No connection"
                        }, fileObj.status)
                    })
                    queue.length = 0
                    queue.push("DONE")
                }
            }
        }
    }



    // handle large files uploading 
    const uploadLargeFile = async (sessionId, fileObj, replaceMap) => {
        //  file select then update progress bar
        // if (fileObj.file._isPlaceholder) {
        //     updateFile(sessionId, filekey, { status: "uploading" })
        // }
        if (fileObj.file._isPlaceholder) {
            updateFile(sessionId, fileObj.filekey, {
                status: "done",
                progress: 100
            }, fileObj.status)
            return; // STOP upload completely
        }

        const { file, filekey, fingerprint, parentId } = fileObj;
        updateFile(sessionId, filekey, { status: "uploading" })

        // create abort controller for this file
        const controller = new AbortController()
        abortControllersRef.current.set(filekey, controller)

        try {

            //  here we are blocking the dangorus file extension 
            if (isBlockedFile(file.name)) {
                updateFile(sessionId, filekey, { status: "blocked", message: "File type is not allowed" })
                return
            }
            const fileHeader = await getFileHeader(file)

            let uploadId, alreadyUploaded

            //  first init data to backend for check status
            const initStart = performance.now()
            const { data: initData } = await axiosApi.post("/upload/init", {
                fileName: file.name,
                fingerprint,
                fileSize: file.size,
                fileType: file.type || "application/octet-stream",
                totalChunks: Math.ceil(file.size / CHUNK_SIZE),
                parent: parentId || null,
                fileHeader,
                replacesFileId: replaceMap?.[file.name] || null
            }, { signal: controller.signal })


            //  if the mime is block from backend 
            if (initData.blocked) {
                console.warn(`[DEBUG] init blocked: ${file.name} | reason=${initData.message}`);
                updateFile(sessionId, filekey, { status: "blocked", message: initData.message })
                return
            }
            console.log(`[INIT] ${(performance.now() - initStart).toFixed(1)}ms | uploadId=${initData.uploadId}`);

            //  if file already uploaded then already uploaded
            if (initData.status === "completed") {
                updateFile(sessionId, filekey, { status: "skipped", progress: 100 }, fileObj.status)
                return
            }


            uploadId = initData.uploadId

            // store uploadId in fileObj for cancle use
            const filesMap = sessionMapsRef.current.get(sessionId)
            if (filesMap?.has(filekey)) {
                filesMap.set(filekey, { ...filesMap.get(filekey), uploadId })
            }


            alreadyUploaded = new Set(initData.uploadedChunks || [])    // how many chunks are already uploaded
            const totalChunks = Math.ceil(file.size / CHUNK_SIZE)

            // here skipp the chunks that are already uploaded
            const chunks = []
            for (let i = 0; i < totalChunks; i++) {
                if (alreadyUploaded.has(i)) continue
                const start = i * CHUNK_SIZE;
                const end = Math.min(start + CHUNK_SIZE, file.size)
                chunks.push({ index: i, start, blob: file.slice(start, end) })
            }

            //  in one batch how many chunk we want
            const batches = []
            for (let i = 0; i < chunks.length; i += CHUNK_BATCH_SIZE) {
                batches.push(chunks.slice(i, i + CHUNK_BATCH_SIZE))
            }

            let uploadCount = alreadyUploaded.size


            //  internet speed tracking here
            let bytesInWindow = 0
            let windowStart = Date.now()
            const SPEED_UPDATE_INTERVAL = 1500     // here every 1.5 second speed update


            //  send batch to the backend /upload-chunk
            console.log(`[CHUNKS] ${file.name} | Total batches: ${batches.length} | Total chunks: ${chunks.length} | Already uploaded: ${alreadyUploaded.size}`)
            for (let i = 0; i < batches.length; i++) {
                const batch = batches[i]
                const batchBytes = batch.reduce((sum, c) => sum + c.blob.size, 0)


                const form = new FormData()
                batch.forEach(c => form.append(`chunk_${c.index}`, c.blob))


                const chunkStart = performance.now()
                console.log(`[BATCH START] ${file.name} | batch ${i + 1}/${batches.length} | chunks: [${batch.map(c => c.index)}] | sending...`)

                let retries = 0
                const MAX_RETRIES = 3
                while (true) {
                    try {
                        await axiosApi.post("/upload/upload-chunk", form, {
                            signal: controller.signal,
                            timeout: 300000,   // 5 miniutes
                            headers: {
                                "x-upload-id": uploadId,
                                "x-indexes": JSON.stringify(batch.map(c => c.index)),
                                "x-starts": JSON.stringify(batch.map(c => c.start))
                            }
                        })
                        console.log(`[BATCH OK] ${file.name} | batch ${i + 1}/${batches.length} | took ${(performance.now() - chunkStart).toFixed(0)}ms`)
                        break // Success!
                    } catch (err) {
                        if (err.name === "CanceledError") return
                        console.error(`[BATCH FAIL DETAIL]`, {
                            fileName: file.name,
                            batchIndex: i,
                            totalBatches: batches.length,
                            errorName: err.name,
                            errorCode: err.code,
                            errorMessage: err.message,
                            httpStatus: err.response?.status,
                            httpData: err.response?.data,
                            isTimeout: err.code === "ECONNABORTED",
                            isNetworkError: !err.response,
                            retriesExhausted: retries >= MAX_RETRIES
                        })

                        const isNetworkError = !err.response
                        if (isNetworkError && retries < MAX_RETRIES) {
                            retries++
                            console.warn(`[RETRY] ${file.name} | batch ${i + 1}/${batches.length} | retry ${retries}/${MAX_RETRIES} | err=${err.message}`)
                            await new Promise(r => setTimeout(r, 2000 * retries)) // Wait 2s, 4s, 6s
                            continue
                        }

                        console.error(`[BATCH FAIL] ${file.name} | batch ${i + 1}/${batches.length} | took ${(performance.now() - chunkStart).toFixed(0)}ms | err.name=${err.name} | err.code=${err.code} | status=${err.response?.status} | msg=${err.message}`)
                        console.error(`[UPLOAD ERROR] ${new Date().toLocaleTimeString()} | File: ${file.name} | Status: ${err.response?.status} | Msg: ${err.message}`);
                        updateFile(sessionId, filekey, { status: "error", message: "Upload failed - connection lost" })
                        throw err
                    }
                }

                // update the speed
                uploadCount += batch.length;
                bytesInWindow += batchBytes


                const now = Date.now()
                const elapsed = now - windowStart

                const pct = parseFloat(((uploadCount / totalChunks) * 100).toFixed(1))

                if (elapsed >= SPEED_UPDATE_INTERVAL) {
                    const speedMBps = (bytesInWindow / (1024 * 1024)) / (elapsed / 1000)
                    updateFile(sessionId, filekey, {
                        progress: pct,
                        speed: speedMBps
                    })

                    // reset window
                    bytesInWindow = 0;
                    windowStart = now
                } else {
                    updateFile(sessionId, filekey, { progress: pct })
                }


            }

            console.log(`[ALL BATCHES DONE] ${file.name} | All ${batches.length} batches finished | calling /upload/complete...`)

            console.log(`[COMPLETE] calling complete for: ${file.name}, uploadId: ${uploadId}`)

            const { data: completeData } = await axiosApi.post("/upload/complete", { uploadId }, { signal: controller.signal })
            updateFile(sessionId, filekey, {
                status: "done",
                progress: 100,
                speed: null,
                id: completeData?.id || null // Store the DB _id
            }, fileObj.status)

            console.log(`[COMPLETE RESPONSE] ${file.name}: ${JSON.stringify(completeData)}`)

            // if (fileObj.isRetry) onUploadComplete?.()


        } catch (error) {
            console.error(`[LARGEFILE CATCH] ${file.name} | error.name=${error.name} | error.code=${error.code} | error.message=${error.message}`)
            if (error.name === "CanceledError") return
            console.error("[DEBUG] uploadLargeFile error response:", error.response?.data);
            console.error("uploadLargeFile error:", error)

            console.error(`[LARGEFILE CATCH DETAIL]`, {
                fileName: file.name,
                errorName: error.name,
                errorCode: error.code,
                errorMessage: error.message,
                httpStatus: error.response?.status,
                httpData: error.response?.data,
                isTimeout: error.code === "ECONNABORTED",
                isNetworkError: !error.response
            })

            // check if already marked as error from chunk failure
            const currentFile = sessionMapsRef.current.get(sessionId)?.get(filekey)
            if (currentFile?.status === "error") {
                console.warn(`[LARGEFILE CATCH] ${file.name} | Already marked as error, skipping`)
                return  // ← already handled, don't overwrite
            }

            const errorMessage = error.response?.data?.message || "Upload failed"
            const isBlocked = error.response?.status === 403 || error.response?.status === 400
            const isNetworkError = !error.response

            console.error(`[LARGEFILE CATCH] ${file.name} | Setting status=${isBlocked ? 'blocked' : 'error'} | msg=${errorMessage}`)
            updateFile(sessionId, filekey, {
                status: isBlocked ? "blocked" : "error",
                message: errorMessage
            }, fileObj.status)

            // if network is down then stop the whole queue (same as small file logic)
            if (isNetworkError) {
                console.error(`[LARGEFILE CATCH] ${file.name} | Network error — stopping entire queue`)
                const queue = uploadQueuesRef.current.get(sessionId)
                if (queue) {
                    queue.forEach(queuedFile => {
                        if (queuedFile === "DONE" || !queuedFile) return
                        updateFile(sessionId, queuedFile.filekey, {
                            status: "error",
                            message: "No connection"
                        }, queuedFile.status)
                    })
                    queue.length = 0
                    queue.push("DONE")
                }
            }
        }

    }


    // create 5 paralle upllading worker here 
    const startUploadWorkers = (sessionId) => {
        if (uploadStartedRef.current.get(sessionId)) return []
        uploadStartedRef.current.set(sessionId, true)

        const queue = uploadQueuesRef.current.get(sessionId)
        console.log(`[WORKERS] Starting ${MAX_CONCURRENT} workers for session ${sessionId} | Queue size: ${queue.length}`)

        return Array(MAX_CONCURRENT).fill(null).map(async (_, workerIndex) => {

            while (true) {

                if (!fileObj) {
                    await new Promise(r => setTimeout(r, 100))
                    continue
                }
                if (fileObj === "DONE") {
                    console.log(`[WORKER-${workerIndex}] saw DONE, stopping`)
                    queue.unshift("DONE")
                    break
                }
                console.log(`[WORKER-${workerIndex}] picking up: ${fileObj.file.name} | isSmall=${fileObj.isSmall} | queue remaining: ${queue.length}`)

                    while (batch.length < SMALL_BATCH_SIZE && queue[0]?.isSmall) {
                        batch.push(queue.shift())
                    }

                console.log(`[WORKER-${workerIndex}] finished processing, looping back for next file`)
            }
            console.log(`[WORKER-${workerIndex}] EXITED the while loop — this worker is now DEAD`)
        })
    }


    //  retru file whe user lcick on retry icon on upload failed here
    const retryFile = (sessionId, filekey) => {
        const filesMap = sessionMapsRef.current.get(sessionId)
        const fileObj = filesMap?.get(filekey)
        if (!fileObj) return

        // reset status
        const updated = { ...fileObj, status: "waiting", progress: 0, speed: null, isRetry: true }
        filesMap.set(filekey, updated)
        updateFile(sessionId, filekey, { status: "waiting", progress: 0, speed: null }, fileObj.status)

        // get queue and remove DONE sentinel first

        if (queue) {

            if (doneIndex !== -1) queue.splice(doneIndex, 1)

            // push file to front
            queue.unshift(updated)

            // add DONE back at end
            queue.push("DONE")
        }

        // reset uploadStarted so workers can restart
        uploadStartedRef.current.set(sessionId, false)

    //  here we are retrying the foder when user clcik on retry icon on folder upload 
    const retryFolder = (sessionId) => {
        const filesMap = sessionMapsRef.current.get(sessionId)
        if (!filesMap) return

        const errorFiles = [];
        filesMap.forEach((fileObj) => {
            if (fileObj.status === "error") {
                errorFiles.push(fileObj)
            }
        })
        console.log(`[DEBUG] retryFolder: Found ${errorFiles.length} files with status 'error'`);
        if (errorFiles.length === 0) return

        // 2. Reset all error files to "waiting"
        errorFiles.forEach(fileObj => {
            // Capture the old "error" status so updateFile knows to subtract from it
            const oldStatus = fileObj.status;
            const oldProgress = fileObj.progress;

            const updated = {
                ...fileObj,
                status: "waiting",

            filesMap.set(fileObj.filekey, updated);

            // Update the UI state - Passing 'oldStatus' ensures the error count drops to 0

            }, oldStatus, oldProgress);
        });

        // 3. RESET QUEUE COMPLETELY
        // We create a fresh queue to avoid any leftovers
        const queue = [];

        // 4. ADD RETRIED FILES BACK TO QUEUE
        errorFiles.forEach(fileObj => {

            queue.push(updated);
        });

        // Add the sentinel so workers know when to stop

        // 5. RESTART WORKERS

//  here for when user click on download file 
export const downloadFile = async (req, res) => {
    try {
        //  file id
        const { id } = req.params;
        const userId = req.user._id;

        //  here check permission for shared file
        const permission = await getUserPermission(userId, id)
        if (!permission) {
            return res.status(403).json({ success: false, message: "Access denied" })
        }


        //  find the file record
        const file = await Upload.findOne({ _id: id, type: "file", uploadStatus: "completed" })
        if (!file) {
            return res.status(404).json({ success: false, message: "File not found" })
        }


        //  now check file exist on disk or not here
        if (!fs.existsSync(file.storagePath)) {
            return res.status(404).json({ success: false, message: "File not found on server" })
        }

        const fileSize = file.fileSize
        //  get range header in bytes from frontned like send 0-4000 bytes 
        const rangeHeader = req.headers.range


        // Use Express built-in helper for headers
        res.attachment(file.name);

        //  tell browser what type of file is this image or SOMETHING ELSE
        res.setHeader("Content-Type", file.fileType || "application/octet-stream")

        // this is for resume which byte range
        res.setHeader("Accept-Ranges", "bytes")


        // if no range header send full file
        if (!rangeHeader) {
            res.setHeader("Content-Length", fileSize)
            //  here sending full file read stream here
            const stream = fs.createReadStream(file.storagePath)
            stream.pipe(res)
            return
        }


        //  if there is a range here so send that data only here
        //  here we recive the bytes from frone edn like 0-999 so convert into [0, 99]
        const parts = rangeHeader.replace(/bytes=/, "").split("-")

        // here get the start byte - 0
        const start = parseInt(parts[0], 10)

        //  get the end byte here - 999 if end byte is not defined here then give rest of the bytes from start byte
        const end = parts[1] ? parseInt(parts[1], 10) : fileSize - 1


        // now validate the bytes range here
        if (start >= fileSize || end >= fileSize) {
            res.setHeader("Content-Range", `bytes */${fileSize}`)
            return res.status(416).json({ success: false, message: "Range is invalid" })
        }


        //  define chunk size
        const chunkSize = end - start + 1

        res.status(206)
        res.setHeader("Content-Range", `bytes ${start}-${end}/${fileSize}`)
        res.setHeader("Content-Length", chunkSize)


        //  now stream only byte range here
        const stream = fs.createReadStream(file.storagePath, { start, end })
        stream.pipe(res)

    } catch (error) {
        res.status(500).json({ success: false, message: error.message })
    }
}




//  when user click on the folder convert into the zip and download it
//   we are creating here child process so our main server is never blocked here
export const downloadFolder = async (req, res) => {
    try {
        const { id } = req.params
        const userId = req.user._id

        // check permission first
        const permission = await getUserPermission(userId, id)
        if (!permission) {
            return res.status(403).json({ success: false, message: "Access denied" })
        }

        // find the folder
        const folder = await Upload.findOne({ _id: id, type: "folder" })
        if (!folder) {
            return res.status(404).json({ success: false, message: "Folder not found" })
        }

        // generate unique zip id for this job
        const zipId = uuidv4()
        const zipPath = path.join(ZIPS_DIR, `${zipId}.zip`)


        //  now in map set intial status is creating 
        zipJobsMap.set(zipId, {
            status: "creating",
            zipPath,
            folderName: folder.name,
            createdAt: Date.now()
        })


        //  resposne to client with zipId
        res.json({ success: true, zipId, folderName: folder.name })


        // so here we are using node js child process this will create zip seperatly
        const child = fork(path.join(__dirname, "../workers/zipWorker.js"))

        // save child process reference in job map
        const currentJob = zipJobsMap.get(zipId)
        if (currentJob) {
            currentJob.childProcess = child
        }

        //  send folder info to the child process
        child.send({ folderId: id.toString(), zipPath })


        //  lisyten message from child here
        child.on("message", (msg) => {
            if (msg.type === "done") {
                //  if zip is already is there update map so down load can find it
                zipJobsMap.set(zipId, {
                    status: "ready",
                    zipPath,
                    folderName: folder.name,
                    fileSize: msg.fileSize,
                    createdAt: Date.now()
                })

                console.log(`[ZIP] Ready: ${zipId} - ${msg.fileSize} bytes`)
            }

            //  if  creating zip errro then 
            if (msg.type === "error") {
                // zip creation failed - update map with error
                zipJobsMap.set(zipId, {
                    status: "error",
                    error: msg.error
                })
                console.error(`[ZIP] Failed: ${zipId} - ${msg.error}`)
            }
        })


        //  if child preocess is creashes 
        child.on("exit", (code) => {
            if (code !== 0) {
                const job = zipJobsMap.get(zipId)
                //  only update if not already mark as done or error
                if (job?.status === "creating") {
                    zipJobsMap.set(zipId, {
                        status: "error",
                        error: "Zip worker crashed"
                    })
                }
            }
        })
        child.on("error", (err) => {
    console.error("[ZIP WORKER ERROR]", err)
})


    } catch (error) {
        res.status(500).json({ success: false, message: error.message })
    }
}

export const downloadZip = async (req, res) => {
    try {
        const { zipId } = req.params

        const job = zipJobsMap.get(zipId)


        //  now here first check if zip is exist and ready
        if (!job || job.status !== "ready") {
            return res.status(404).json({ success: false, message: "Zip not ready or not found" })
        }


        // check zip file actually exists on disk
        if (!fs.existsSync(job.zipPath)) {
            return res.status(404).json({ success: false, message: "Zip file not found on disk" })
        }


        const fileSize = job.fileSize;

        // get range header in bytes from frontned like send 0-4000 bytes 
        const rangeHeader = req.headers.range


        // set headers same as downloadFile
        res.setHeader("Content-Disposition", `attachment; filename="${job.folderName}.zip"`)
        res.setHeader("Content-Type", "application/zip")
        res.setHeader("Accept-Ranges", "bytes")


        //  no range header - send null
        if (!rangeHeader) {
            res.setHeader("Content-Length", fileSize)
            const stream = fs.createReadStream(job.zipPath)
            stream.pipe(res)
            return
        }

        // parse range header - identical to downloadFile
        const parts = rangeHeader.replace(/bytes=/, "").split("-")
        const start = parseInt(parts[0], 10)
        const end = parts[1] ? parseInt(parts[1], 10) : fileSize - 1


        // validate range
        if (start >= fileSize || end >= fileSize) {
            res.setHeader("Content-Range", `bytes */${fileSize}`)
            return res.status(416).json({ success: false, message: "Range is invalid" })
        }

        const chunkSize = end - start + 1

        res.status(206)
        res.setHeader("Content-Range", `bytes ${start}-${end}/${fileSize}`)
        res.setHeader("Content-Length", chunkSize)

        // stream only requested byte range - identical to downloadFile
        const stream = fs.createReadStream(job.zipPath, { start, end })
        stream.pipe(res)


    } catch (error) {
        res.status(500).json({ success: false, message: error.message })
    }
}





// delete zip file form storage when download is comple in frontend
export const deleteZip = async (req, res) => {
    try {
        const { zipId } = req.params

        const job = zipJobsMap.get(zipId)
        if (!job) {
            return res.status(404).json({ success: false, message: "Zip job not found" })
        }

        // Kill the child process if it's still running
        if (job.status === "creating" && job.childProcess) {
            job.childProcess.kill()
            console.log(`[ZIP] Killed worker process for zip: ${zipId}`)
        }

        // delete zip file from disk
        if (job.zipPath && fs.existsSync(job.zipPath)) {
            try {
                fs.unlinkSync(job.zipPath)
                console.log(`[ZIP] Deleted: ${zipId}`)
            } catch (err) {
                console.error(`[ZIP] Delete failed: ${zipId}`, err.message)
            }
        }

        // remove from map
        zipJobsMap.delete(zipId)

        res.json({ success: true })


    } catch (error) {
        res.status(500).json({ success: false, message: error.message })

    }
}

// Get folder files and paths metadata recursively (BFS)
export const getFolderMetadata = async (req, res) => {
    try {
        const { id } = req.params
        const userId = req.user._id

        // check permission first
        const permission = await getUserPermission(userId, id)
        if (!permission) {
            return res.status(403).json({ success: false, message: "Access denied" })
        }

        // find the folder
        const folder = await Upload.findOne({ _id: id, type: "folder" })
        if (!folder) {
            return res.status(404).json({ success: false, message: "Folder not found" })
        }

        // BFS to fetch all descendants
        let parentIds = [folder._id]
        const allItems = []

        while (parentIds.length > 0) {
            const children = await Upload.find({
                parent: { $in: parentIds },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            }).select("name type parent fileSize fileType").lean()

            allItems.push(...children)

            // next level: only look inside folders
            parentIds = children
                .filter(c => c.type === "folder")
                .map(c => c._id)
        }

        // Build path map (folderId -> relativePath)
        const pathMap = new Map()
        pathMap.set(id.toString(), "")

        const folders = allItems.filter(i => i.type === "folder")
        for (const f of folders) {
            const parentPath = pathMap.get(f.parent.toString()) || ""
            pathMap.set(f._id.toString(), path.join(parentPath, f.name))
        }

        // Map files with relative path
        const files = allItems.filter(i => i.type === "file").map(file => {
            const parentPath = pathMap.get(file.parent.toString()) || ""
            return {
                id: file._id,
                name: file.name,
                fileSize: file.fileSize,
                fileType: file.fileType,
                relativePath: path.join(parentPath, file.name).replace(/\\/g, "/")
            }
        })

        res.json({
            success: true,
            folderName: folder.name,
            files
        })

    } catch (error) {
        res.status(500).json({ success: false, message: error.message })
    }
}

                    if (abortController.signal.aborted) break

                    const chunk = queue.shift()
                    if (!chunk) break

                    try {
                        const response = await axiosApi.get(endpoint, {
                            headers: { Range: `bytes=${chunk.start}-${chunk.end}` },
                            responseType: "arraybuffer",
                            signal: abortController.signal
                        })

                        if (abortController.signal.aborted) break

                        await saveChunk(db, fileId, chunk.index, response.data)

                        downloadedCount++
                        bytesInWindow += chunk.end - chunk.start + 1

                        const now = Date.now()
                        const elapsed = now - windowStart
                        const progress = parseFloat(((downloadedCount / totalChunks) * 100).toFixed(1))

                        const sessionKey = sessionId

                        if (elapsed >= SPEED_UPDATE_INTERVAL) {
                            const speedMBps = (bytesInWindow / (1024 * 1024)) / (elapsed / 1000)
                            updateSession(sessionKey, { progress, speed: speedMBps })
                            bytesInWindow = 0
                            windowStart = now
                        } else {
                            updateSession(sessionKey, { progress })
                        }

                        await new Promise(r => setTimeout(r, 100))

                    } catch (error) {
                        if (error.name === "CanceledError") return
                        console.warn(`chunk ${chunk.index} failed, retrying...`, error)
                        queue.push(chunk)
                    }
                }
            }

            // run concurrent workers
            await Promise.all(Array(MAX_CONCURRENT).fill(null).map(() => worker()))

            // if finished without abort - assemble and save
            if (!abortController.signal.aborted) {
                const sessionKey = sessionId
                updateSession(sessionKey, { status: "assembling", progress: 100 })

                // Hand off to Service Worker for persistent streaming
                await assembleAndSave(db, fileId, totalChunks, fileName, fileType, fileSize, sessionId)

                // We don't cleanupChunks immediately because the SW is still reading them!
                // The cleanup should happen once the SW is done or the user closes the session.
                // For now, let's leave it to be cleaned up when the session is closed.
            }

        } catch (error) {
            console.error("downloadInChunks error:", error)
            const sessionKey = sessionId
            updateSession(sessionKey, { status: "error" })
        } finally {
            activeDownloadsRef.current.delete(fileId)
        }
    }


    // -------------------------------------------------------
    // YOUR EXISTING downloadFile - UNTOUCHED
    // just calls the shared downloadInChunks function now
    // -------------------------------------------------------
    const downloadFile = async (file) => {
        const { _id: fileId, name, fileSize, fileType } = file

        if (activeDownloadsRef.current.has(fileId)) return

        const db = await initDB()
        const totalChunks = Math.ceil(fileSize / CHUNK_SIZE)
        const savedChunks = await getSavedChunks(db, fileId, totalChunks)

        const progress = parseFloat(((savedChunks.size / totalChunks) * 100).toFixed(1))

        const sessionId = crypto.randomUUID()

        setSessions(prev => [{
            id: sessionId,
            fileId,
            name,
            fileSize,
            fileType,
            totalChunks,
            status: "downloading",
            progress,
            speed: null,
            error: null
        }, ...prev])

        setIsPanelOpen(true)
        setIsMinimized(false)

        await downloadInChunks(
            fileId,
            `/download/file/${fileId}`,
            fileSize,
            name,
            fileType,
            false,
            null,
            sessionId
        )
    }


    // -------------------------------------------------------
    // NEW downloadFolder - completely rewritten
    // Phase 1 - create zip on backend and poll until ready
    // Phase 2 - download zip in chunks exactly like a file
    // resume works because zipId and fileSize saved to IndexedDB
    // -------------------------------------------------------
    const downloadFolder = async (folder) => {
        const { _id: folderId, name } = folder

        // check if already downloading this folder
        if (activeDownloadsRef.current.has(folderId)) return

        const sessionId = crypto.randomUUID()

        // add session to panel immediately
        setSessions(prev => [{
            id: sessionId,
            fileId: folderId,
            name: `${name}.zip`,
            fileSize: null,
            status: "creating",   // new status - zip is being created
            progress: 0,
            speed: null,
            isFolder: true
        }, ...prev])

        setIsPanelOpen(true)
        setIsMinimized(false)

        try {
            const db = await initDB()

            // -----------------------------------------------
            // CHECK INDEXEDDB FIRST - resume after refresh
            // if zipJob exists in IndexedDB zip was already
            // created before refresh so skip phase 1 entirely
            // -----------------------------------------------
            let zipId, fileSize, folderName

            const savedZipJob = await getZipJob(db, folderId)

            if (savedZipJob) {
                // zip was already created before refresh
                // jump straight to chunk download phase
                console.log(`[FOLDER DOWNLOAD] Resuming from saved zip job: ${savedZipJob.zipId}`)
                zipId = savedZipJob.zipId
                fileSize = savedZipJob.fileSize
                folderName = savedZipJob.folderName

                // update session with correct fileSize and status
                updateSession(sessionId, {
                    status: "downloading",
                    fileSize,
                    name: `${folderName}.zip`,
                    zipId
                })

            } else {
                // -----------------------------------------------
                // PHASE 1 - create zip on backend
                // -----------------------------------------------
                updateSession(sessionId, { status: "creating" })

                // call backend to start zip creation
                // backend responds immediately with zipId
                // child process builds zip in background
                const { data: createData } = await axiosApi.post(`/download/folder/${folderId}`)

                if (!createData.success) {
                    updateSession(sessionId, { status: "error" })
                    return
                }

                zipId = createData.zipId
                folderName = createData.folderName

                // Store zipId in session immediately so it can be canceled during creation
                updateSession(sessionId, { zipId })

                // -----------------------------------------------
                // PHASE 1b - poll backend every 2 seconds
                // until zip status is "ready" or "error"
                // -----------------------------------------------
                fileSize = await new Promise((resolve, reject) => {
                    const interval = setInterval(async () => {
                        try {
                            const { data: statusData } = await axiosApi.get(`/download/zip-status/${zipId}`)

                            if (statusData.status === "ready") {
                                // zip is ready on disk
                                clearInterval(interval)
                                pollingIntervalsRef.current.delete(folderId)
                                resolve(statusData.fileSize)
                            }

                            if (statusData.status === "error") {
                                clearInterval(interval)
                                pollingIntervalsRef.current.delete(folderId)
                                reject(new Error(statusData.error || "Zip creation failed"))
                            }

                        } catch (err) {
                            clearInterval(interval)
                            pollingIntervalsRef.current.delete(folderId)
                            reject(err)
                        }
                    }, 2000) // poll every 2 seconds

                    // save interval id so we can cancel it if user cancels
                    pollingIntervalsRef.current.set(folderId, interval)
                })

                // -----------------------------------------------
                // SAVE ZIP JOB TO INDEXEDDB before starting download
                // so if browser refreshes during download
                // we can skip zip creation and resume chunks directly
                // -----------------------------------------------
                await saveZipJob(db, folderId, { zipId, fileSize, folderName })

                // update session now that we know the file size
                updateSession(sessionId, {
                    status: "downloading",
                    fileSize,
                    name: `${folderName}.zip`,
                    zipId
                })
            }

            // -----------------------------------------------
            // PHASE 2 - download zip in chunks
            // zipId is used as fileId for chunk storage in IndexedDB
            // this is identical to how file download works
            // -----------------------------------------------
            await downloadInChunks(
                zipId,                          // used as fileId for IndexedDB chunk keys
                `/download/zip/${zipId}`,       // endpoint with range request support
                fileSize,
                `${folderName}.zip`,
                "application/zip",
                true,                           // isFolder flag
                folderId,                       // used as session key in panel
                sessionId
            )

        } catch (error) {
            console.error("downloadFolder error:", error)
            updateSession(sessionId, { status: "error" })
            // clear polling if still running
            const interval = pollingIntervalsRef.current.get(folderId)
            if (interval) {
                clearInterval(interval)
                pollingIntervalsRef.current.delete(folderId)
            }
        }
    }

    // -------------------------------------------------------
    // NEW downloadFolderClientSide - client side zipping
    // -------------------------------------------------------
    const downloadFolderClientSide = async (folder) => {
        const { _id: folderId, name } = folder

        // check if already downloading this folder
        if (activeDownloadsRef.current.has(folderId)) return

        const sessionId = crypto.randomUUID()
        const abortController = new AbortController()
        activeDownloadsRef.current.set(folderId, abortController)

        // add session to panel immediately
        setSessions(prev => [{
            id: sessionId,
            fileId: folderId,
            name: `${name}.zip`,
            fileSize: null,
            status: "downloading",
            progress: 0,
            speed: null,
            isFolder: true,
            isClientZip: true
        }, ...prev])

        setIsPanelOpen(true)
        setIsMinimized(false)

        try {
            const db = await initDB()

            // Fetch folder metadata
            const { data } = await axiosApi.get(`/download/folder-metadata/${folderId}`, {
                signal: abortController.signal
            })

            const { folderName, files } = data
            const totalFiles = files.length
            const totalBytes = files.reduce((sum, f) => sum + f.fileSize, 0)

            // Update session with correct file size
            updateSession(sessionId, { fileSize: totalBytes })

            // Check if there is already a saved job in IndexedDB
            let savedJob = await db.get(ZIP_JOB_STORE, `clientzip_${folderId}`)
            if (!savedJob) {
                savedJob = {
                    folderId,
                    folderName,
                    files,
                    downloadedFiles: {},
                    sessionId
                }
                await db.put(ZIP_JOB_STORE, savedJob, `clientzip_${folderId}`)
            }

            // Speed and progress tracking in memory
            let bytesInWindow = 0
            let windowStart = Date.now()
            const SPEED_UPDATE_INTERVAL = 1500

            // Initialize totalDownloadedBytes from already downloaded files / chunks
            let totalDownloadedBytes = 0
            for (const file of files) {
                if (savedJob.downloadedFiles[file.id]) {
                    totalDownloadedBytes += file.fileSize
                } else {
                    const fileTotalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                    const savedChunks = await getSavedChunks(db, file.id, fileTotalChunks)
                    for (const chunkIndex of savedChunks) {
                        const start = chunkIndex * CHUNK_SIZE
                        const end = Math.min(start + CHUNK_SIZE, file.fileSize) - 1
                        totalDownloadedBytes += (end - start + 1)
                    }
                }
            }

            // Worker function to process the file queue
            const fileQueue = [...files]
            const activeWorkers = []
            const maxFileWorkers = 2

            const downloadFileWorker = async () => {
                while (fileQueue.length > 0 && !abortController.signal.aborted) {
                    const file = fileQueue.shift()
                    if (!file) break

                    // check if already downloaded
                    if (savedJob.downloadedFiles[file.id]) {
                        continue
                    }

                    const fileTotalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                    const fileSavedChunks = await getSavedChunks(db, file.id, fileTotalChunks)

                    // build chunk download list
                    const chunksToDownload = []
                    for (let i = 0; i < fileTotalChunks; i++) {
                        if (!fileSavedChunks.has(i)) {
                            const start = i * CHUNK_SIZE
                            const end = Math.min(start + CHUNK_SIZE, file.fileSize) - 1
                            chunksToDownload.push({ index: i, start, end })
                        }
                    }

                    // Download chunks for this file
                    let success = true
                    while (chunksToDownload.length > 0 && !abortController.signal.aborted) {
                        const chunk = chunksToDownload.shift()
                        try {
                            const response = await axiosApi.get(`/download/file/${file.id}`, {
                                headers: { Range: `bytes=${chunk.start}-${chunk.end}` },
                                responseType: "arraybuffer",
                                signal: abortController.signal
                            })
                            await saveChunk(db, file.id, chunk.index, response.data)

                            totalDownloadedBytes += chunk.end - chunk.start + 1
                            bytesInWindow += chunk.end - chunk.start + 1
                            const now = Date.now()
                            const elapsed = now - windowStart
                            const progress = parseFloat(((totalDownloadedBytes / totalBytes) * 100).toFixed(1))

                            if (elapsed >= SPEED_UPDATE_INTERVAL) {
                                const speedMBps = (bytesInWindow / (1024 * 1024)) / (elapsed / 1000)
                                updateSession(sessionId, { progress, speed: speedMBps })
                                bytesInWindow = 0
                                windowStart = now
                            } else {
                                updateSession(sessionId, { progress })
                            }
                        } catch (error) {
                            if (error.name === "CanceledError") {
                                success = false
                                break
                            }
                            console.warn(`File ${file.name} chunk ${chunk.index} failed, retrying...`, error)
                            chunksToDownload.push(chunk)
                            await new Promise(r => setTimeout(r, 1000))
                        }
                    }

                    if (success && !abortController.signal.aborted) {
                        savedJob.downloadedFiles[file.id] = true
                        await db.put(ZIP_JOB_STORE, savedJob, `clientzip_${folderId}`)
                    }
                }
            }

            // Spawn concurrent file workers
            for (let i = 0; i < Math.min(maxFileWorkers, fileQueue.length); i++) {
                activeWorkers.push(downloadFileWorker())
            }

            await Promise.all(activeWorkers)

            if (abortController.signal.aborted) return

            // Phase 3: Assembly
            updateSession(sessionId, { status: "assembling", progress: 100 })

            // Build zip input list
            const zipInputs = []
            for (const file of files) {
                const fileTotalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                
                // Create readable stream for the file from IndexedDB chunks
                const fileStream = new ReadableStream({
                    async start(controller) {
                        try {
                            // Start prefetching chunk 0
                            let nextChunkPromise = db.get(STORE_NAME, `${file.id}_chunk_0`)

                            for (let i = 0; i < fileTotalChunks; i++) {
                                const chunk = await nextChunkPromise
                                if (!chunk) {
                                    controller.error(new Error(`Missing chunk ${i} for file ${file.name}`))
                                    return
                                }

                                // Start prefetching chunk i+1 in parallel
                                if (i + 1 < fileTotalChunks) {
                                    nextChunkPromise = db.get(STORE_NAME, `${file.id}_chunk_${i + 1}`)
                                }

                                controller.enqueue(new Uint8Array(chunk))
                            }
                            controller.close()
                        } catch (err) {
                            controller.error(err)
                        }
                    }
                })

                zipInputs.push({
                    name: file.relativePath,
                    input: fileStream
                })
            }

            // Stream Zip to Disk
            const zipResponse = downloadZip(zipInputs)
            const fileStream = streamSaver.createWriteStream(`${folderName}.zip`, {
                size: zipResponse.headers.get("Content-Length") || undefined
            })
            await zipResponse.body.pipeTo(fileStream)

            // Cleanup chunks and state from IndexedDB
            for (const file of files) {
                const fileTotalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                await cleanupChunks(db, file.id, fileTotalChunks)
            }
            await db.delete(ZIP_JOB_STORE, `clientzip_${folderId}`)

            updateSession(sessionId, { status: "done", progress: 100 })

        } catch (error) {
            console.error("Client-side downloadFolder error:", error)
            updateSession(sessionId, { status: "error" })
        } finally {
            activeDownloadsRef.current.delete(folderId)
        }
    }


    // -------------------------------------------------------
    // abortDownload - clears active download and polling interval
    // -------------------------------------------------------
    const abortDownload = (fileId) => {
        const controller = activeDownloadsRef.current.get(fileId)
        if (controller) {
            controller.abort()
            activeDownloadsRef.current.delete(fileId)
        }
        // if it's a folder, also clear the zip-status polling interval
        const interval = pollingIntervalsRef.current.get(fileId)
        if (interval) {
            clearInterval(interval)
        }
    }

    // -------------------------------------------------------
    // closeSession - updated to handle folder zip cleanup
    // -------------------------------------------------------
    const closeSession = async (sessionId) => {
        const session = sessions.find(s => s.id === sessionId)
        if (!session) return;
        const fileId = session.fileId;

        // stop download immediately
        abortDownload(fileId)

        const db = await initDB()

        if (session && !session.isFolder) {
            // file - cleanup chunks from IndexedDB
            try {
                await cleanupChunks(db, fileId, session.totalChunks)
            } catch (e) {
                console.error("Chunk cleanup failed:", e)
            }
        }

        if (session && session.isClientZip) {
            // client-side zipped folder download
            try {
                const savedJob = await db.get(ZIP_JOB_STORE, `clientzip_${fileId}`)
                if (savedJob) {
                    for (const file of savedJob.files) {
                        const totalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                        await cleanupChunks(db, file.id, totalChunks)
                    }
                    await db.delete(ZIP_JOB_STORE, `clientzip_${fileId}`)
                }
            } catch (e) {
                console.error("Client zip cleanup failed:", e)
            }
        }

        if (session && session.isFolder && !session.isClientZip) {
            // folder - cleanup zip job from IndexedDB
            // and delete zip from backend disk
            try {
                const zipJob = await getZipJob(db, fileId)
                if (zipJob) {
                    // delete chunks saved for this zip
                    const totalChunks = Math.ceil(zipJob.fileSize / CHUNK_SIZE)
                    await cleanupChunks(db, zipJob.zipId, totalChunks)
                    // delete zip job record
                    await deleteZipJob(db, fileId)
                    // tell backend to delete zip file from disk
                    await axiosApi.delete(`/download/zip/${zipJob.zipId}`)
                } else if (session.zipId) {
                    // If canceled during Phase 1 (zipping in progress)
                    await axiosApi.delete(`/download/zip/${session.zipId}`)
                }
            } catch (e) {
                console.error("Zip cleanup failed:", e)
            }
        }

        setSessions(prev => {
            const updated = prev.filter(s => s.id !== sessionId)
            if (updated.length === 0) setIsPanelOpen(false)
            return updated
        })
    }


    // -------------------------------------------------------
    // closeAllSessions - updated to handle folders too
    // -------------------------------------------------------
    const closeAllSessions = async () => {
        // stop all downloads immediately
        activeDownloadsRef.current.forEach((controller) => controller.abort())
        activeDownloadsRef.current.clear()

        // clear all polling intervals
        pollingIntervalsRef.current.forEach((interval) => clearInterval(interval))
        pollingIntervalsRef.current.clear()

        try {
            const db = await initDB()
            for (const session of sessions) {
                if (!session.isFolder) {
                    // cleanup file chunks
                    await cleanupChunks(db, session.fileId, session.totalChunks)
                } else if (session.isClientZip) {
                    // cleanup client-side zip job and chunks
                    const savedJob = await db.get(ZIP_JOB_STORE, `clientzip_${session.fileId}`)
                    if (savedJob) {
                        for (const file of savedJob.files) {
                            const totalChunks = Math.ceil(file.fileSize / CHUNK_SIZE)
                            await cleanupChunks(db, file.id, totalChunks)
                        }
                        await db.delete(ZIP_JOB_STORE, `clientzip_${session.fileId}`)
                    }
                } else {
                    // cleanup folder zip job and chunks
                    const zipJob = await getZipJob(db, session.fileId)
                    if (zipJob) {
                        const totalChunks = Math.ceil(zipJob.fileSize / CHUNK_SIZE)
                        await cleanupChunks(db, zipJob.zipId, totalChunks)
                        await deleteZipJob(db, session.fileId)
                        // tell backend to delete zip file from disk
                        try {
                            await axiosApi.delete(`/download/zip/${zipJob.zipId}`)
                        } catch (err) {
                            console.error("deleteZip failed:", err.message)
                        }
                    } else if (session.zipId) {
                        // If canceled during Phase 1 (zipping in progress)
                        try {
                            await axiosApi.delete(`/download/zip/${session.zipId}`)
                        } catch (err) {
                            console.error("deleteZip failed:", err.message)
                        }
                    }
                }
            }
        } catch (e) {
            console.error("Cleanup all failed:", e)
        }

        setSessions([])
        setIsPanelOpen(false)
    }

    const toggleMinimized = () => {
        setIsMinimized(prev => !prev)
    }

    // Auto resume client-side folder downloads on mount
    useEffect(() => {
        const resumeClientZips = async () => {
            try {
                const db = await initDB()
                const tx = db.transaction(ZIP_JOB_STORE, "readonly")
                const store = tx.store
                let cursor = await store.openCursor()
                while (cursor) {
                    if (cursor.key.startsWith("clientzip_")) {
                        const job = cursor.value
                        console.log(`[CLIENT ZIP RESUME] Resuming job for folder: ${job.folderId}`)
                        downloadFolderClientSide({ _id: job.folderId, name: job.folderName })
                    }
                    cursor = await cursor.continue()
                }
            } catch (err) {
                console.error("Failed to auto-resume client zips:", err)
            }
        }
        resumeClientZips()
    }, [])

    return (
        <DownloadContext.Provider value={{
            sessions,
            isPanelOpen,
            isMinimized,
            toggleMinimized,
            downloadFile,
            downloadFolder,
            downloadFolderClientSide,
            closeSession,
            closeAllSessions
        }}>
            {children}
        </DownloadContext.Provider>
    )
}

export function useDownload() {
    return useContext(DownloadContext)
}

import React, { useRef, useState, useEffect, memo } from "react";
import InteractiveIcon from "../layout/InteractiveIcon";
import { Dropdown } from "react-bootstrap";
import arrowRight from "@images/icon/arrow-right.svg";
import folderPlus from "@images/icon/folder-plus.svg";
import userPlus from "@images/icon/user-plus.svg";
import downloadIcon from "@images/icon/download.svg";
import renameIcon from "@images/icon/rename.svg";
import moveIcon from "@images/icon/move.svg";
import trash from "@images/icon/trash.svg";
import menuDotsOutlineIcon from "@images/icon/menu-dots-outline.svg";
import copyIcon from "@images/icon/copy.svg";
import addFileIcon from "@images/icon/file-plus.svg";
import folderUploadIcon from "@images/icon/folder-upload.svg";
import listFolder9Icon from "@images/svgs/list/SF9.svg";
import colorIcon from "@images/icon/color.svg";


const Breadcrumbs = memo(function Breadcrumbs({
    trail = [],
    onNavigate,
    onHomeClick,
    maxVisible = 5,
    rootLabel = "Home",
    actions = [],
    setModal,
    selectedIds,
    items,
    currentFolderId,
    addFiles,
    downloadFile,
    downloadFolder,
    onNewFolder,
    changeColor,
}) {

    //  here actual function starts
    const [showActionDropdown, setShowActionDropdown] = useState(false)
    const [showColorDropdown, setShowColorDropdown] = useState(false)
    const dropdownRef = useRef(null)
    const fileInputRef = useRef(null)
    const folderInputRef = useRef(null)


    // how many ids is selected in check box
    const selectedArray = selectedIds ? Array.from(selectedIds) : [];
    const selectedItem = items?.find(i => i._id === selectedArray[0])
    const hasFolder = selectedArray.some(id => items?.find(i => i._id === id)?.type === "folder")
    
    // Robust check for both Set and Array
    const selectedCount = selectedIds?.size ?? selectedIds?.length ?? 0;
    const isDisabled = selectedCount === 0;

    //  when file uploading
    const handleFileChange = (e) => {
        const files = Array.from(e.target.files);
        if (files.length) {
            addFiles(files, currentFolderId)
            e.target.value = ""
        }
    }

    // when folder upload
    const handleUploadFolder = () => folderInputRef.current.click()
    const handleAddFiles = () => fileInputRef.current.click()


    //  here when user download
    const handleDownload = () => {
        selectedArray.forEach(id => {
            const item = items?.find(i => i._id === id)
            if (!item) return;
            if (item.type === "file") downloadFile(item)
            else downloadFolder(item)
        })
    }



    //  when user creates new folder here
    const handleNewFolder = () => {
        if (onNewFolder) {
            onNewFolder()
        } else {
            setModal({ type: "createNewFolder" })
        }
        setShowActionDropdown(false)
    }


    //  when rename modal
    const handleRename = () => {
    if (!selectedItem) return
    setModal({ type: "RenameModal", data: selectedItem })
    setShowActionDropdown(false)
}


    //  when user share 
    const handleShare = () => {
        if (!selectedItem) return;
        setModal({ type: "shareUser", data: selectedItem });
        setShowActionDropdown(false);
    };


    //  file folder moving
    const handleMove = () => {
        if (isDisabled) return;
        setModal({ type: "MoveModal", data: selectedArray });
        setShowActionDropdown(false);
    };


    //  file folder copy
    const handleCopy = () => {
        if (isDisabled) return;
        setModal({ type: "CopyModal", data: selectedArray });
        setShowActionDropdown(false);
    };


    //  file folder delete
    const handleDelete = () => {
        if (isDisabled) return;
        setModal({ type: "DeleteModal", data: selectedArray });
        setShowActionDropdown(false);
    };


    //  when user click out side of modal close it
    useEffect(() => {
        const handleClickOutside = (event) => {
            if (dropdownRef.current && !dropdownRef.current.contains(event.target)) {
                setShowActionDropdown(false);
            }
        };
        document.addEventListener("mousedown", handleClickOutside);
        return () => document.removeEventListener("mousedown", handleClickOutside);
    }, []);
    
    // reset color dropdown state if parent actions dropdown is closed
    useEffect(() => {
        if (!showActionDropdown) {
            setShowColorDropdown(false);
        }
    }, [showActionDropdown]);



    // 
    const shouldCollapse = trail.length > maxVisible;
    const collapsedItems = shouldCollapse ? trail.slice(0, trail.length - maxVisible) : [];
    const visibleTrail = shouldCollapse ? trail.slice(trail.length - maxVisible) : trail;


    // here all menu items like new folder share move all here 
    const menuItems = actions.length > 0 ? (
        <Dropdown.Menu className="custom-breadcrumb-dropdown" style={{ overflow: "visible" }}>
            {actions.includes("newFolder") && (
                <Dropdown.Item as="button" onClick={handleNewFolder}>
                    <span className="d-flex align-items-center">
                        <InteractiveIcon defaultIcon={folderPlus} className="me-2" width={20} />
                        New Folder
                    </span>
                </Dropdown.Item>
            )}

            {actions.includes("uploadFolder") && (
                <Dropdown.Item as="button" onClick={handleUploadFolder}>
                    <span className="d-flex align-items-center">
                        <InteractiveIcon defaultIcon={folderUploadIcon} className="me-2" width={20} />
                        Upload Folder
                    </span>
                </Dropdown.Item>
            )}

            {actions.includes("addFiles") && (
                <Dropdown.Item as="button" onClick={handleAddFiles}>
                    <span className="d-flex align-items-center">
                        <InteractiveIcon defaultIcon={addFileIcon} className="me-2" width={20} />
                        Add Files
                    </span>
                </Dropdown.Item>
            )}

            {actions.includes("share") && (
                <Dropdown.Item as="button" disabled={isDisabled} onClick={handleShare} style={{ opacity: isDisabled ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={userPlus} className="me-2" width={20} /> Share
                </Dropdown.Item>
            )}

            {actions.includes("download") && (
                <Dropdown.Item as="button" disabled={isDisabled} onClick={handleDownload} style={{ opacity: isDisabled ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={downloadIcon} className="me-2" width={20} /> Download
                </Dropdown.Item>
            )}

            {actions.includes("rename") && (
                <Dropdown.Item as="button" disabled={selectedCount !== 1} onClick={handleRename} style={{ opacity: selectedCount !== 1 ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={renameIcon} className="me-2" width={20} /> Rename
                </Dropdown.Item>
            )}

            {actions.includes("changeColor") && (
                <div className="position-relative w-100">
                    <button 
                        className="dropdown-item d-flex align-items-center justify-content-between" 
                        disabled={!hasFolder} 
                        style={{ opacity: !hasFolder ? 0.5 : 1, width: "100%" }}
                        onClick={(e) => {
                            e.stopPropagation();
                            if (hasFolder) setShowColorDropdown(prev => !prev);
                        }}
                    >
                        <span className="d-flex align-items-center">
                            <InteractiveIcon defaultIcon={colorIcon} className="me-2" width={20} /> Change Color
                        </span>
                    </button>

                    {hasFolder && showColorDropdown && (
                        <div 
                            className="show position-absolute" 
                            style={{ 
                                zIndex: 10000, 
                                left: "100%", 
                                top: 0,
                                minWidth: "226px",
                                maxWidth: "226px",
                                padding: "20px",
                                background: "var(--white)",
                                border: "1px solid var(--secondary)",
                                borderRadius: "8px",
                                boxShadow: "0px 4px 24px 0px rgba(0, 0, 0, 0.10)",
                            }}
                        >
                            <p className="mb-3" style={{ fontSize: "14px", color: "var(--dark-50)" }}>Folder Color </p>
                            <div className="d-flex align-items-center flex-wrap" style={{ margin: "-8px", marginTop: "-8px" }}>

                                    <button 
                                        key={color}
                                        className="border-0"
                                        style={{ 
                                            position: "relative",
                                            display: "block",
                                            width: "24px",
                                            height: "24px",
                                            borderRadius: "50%",
                                            margin: "8px",
                                            outline: "1px solid var(--dark-20)",
                                            outlineOffset: "-1px",
                                            padding: 0,
                                            cursor: "pointer",
                                            backgroundColor: `var(--${color})`
                                        }}
                                        onClick={(e) => { 
                                            e.stopPropagation(); 
                                            changeColor?.(selectedArray, color); 
                                            setShowActionDropdown(false); 
                                        }}
                                    />
                                ))}
                            </div>
                        </div>
                    )}
                </div>
            )}

            {actions.includes("copy") && (
                <Dropdown.Item as="button" disabled={isDisabled} onClick={handleCopy} style={{ opacity: isDisabled ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={copyIcon} className="me-2" width={20} /> Copy
                </Dropdown.Item>
            )}

            {actions.includes("move") && (
                <Dropdown.Item as="button" disabled={isDisabled} onClick={handleMove} style={{ opacity: isDisabled ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={moveIcon} className="me-2" width={20} /> Move
                </Dropdown.Item>
            )}

            {actions.includes("trash") && (
                <Dropdown.Item as="button" disabled={isDisabled} onClick={handleDelete} style={{ opacity: isDisabled ? 0.5 : 1 }}>
                    <InteractiveIcon defaultIcon={trash} className="me-2" width={20} /> Trash
                </Dropdown.Item>
            )}
        </Dropdown.Menu>

    return (

                <>
                    <input type="file" ref={fileInputRef} hidden multiple onChange={handleFileChange} />
                    <input type="file" ref={folderInputRef} hidden webkitdirectory="true" multiple onChange={handleFileChange} />
                </>
            ) : null}


            {/*  actual breadCumb here */}
            <ul className="gap-1 breadcrumb mb-0">

                {/*  root */}
                <li className="position-relative" ref={trail.length === 0 ? dropdownRef : null}>
                    <a
                        className={`cursor-pointer ${trail.length === 0 ? "highlight" : ""}`}
                        onClick={() => {
                            if (trail.length === 0 && actions.length > 0) {
                                setShowActionDropdown(prev => !prev)
                            } else {
                                onHomeClick?.();
                                setShowActionDropdown(false)
                            }
                        }}>
                        <span>{rootLabel}</span>
                    </a>
                    {trail.length === 0 && actions.length > 0 && showActionDropdown && (
                        <Dropdown show>{menuItems}</Dropdown>
                    )}
                </li>


                {/* COLLAPSED  */}
                {shouldCollapse && (
                    <>
                        <li className="d-flex align-items-center breadcrumb-indicator">
                            <InteractiveIcon defaultIcon={arrowRight} width={16} height={16} />
                        </li>
                        <li className="d-flex align-items-center">
                            <a className="over-breadcrumb-links-folder p-0">
                                <Dropdown>
                                    <Dropdown.Toggle className="no-border-btn">
                                        <InteractiveIcon defaultIcon={menuDotsOutlineIcon} width={24} />

                                    {/*  drop down show like all folder name here */}
                                    <Dropdown.Menu>
                                        {collapsedItems.map((folder, index) => (

                                    </Dropdown.Menu>

                {/* here viible trail like > foldername > foldername */}
                {visibleTrail.map((folder, index) => {

import { useState, useMemo, useRef, useEffect } from "react"
import { Document, Page, pdfjs } from "react-pdf"
import { Virtuoso } from "react-virtuoso"
import "react-pdf/dist/Page/AnnotationLayer.css"
import "react-pdf/dist/Page/TextLayer.css"

pdfjs.GlobalWorkerOptions.workerSrc = new URL(
    "pdfjs-dist/build/pdf.worker.min.mjs",
    import.meta.url
).toString()

function PdfViewer({ file: fileData }) {
    const BASE_URL = import.meta.env.VITE_API_URL
    const fileUrl = `${BASE_URL}/files${fileData.storagePath}`

    const [numPages, setNumPages] = useState(null)
    const [loadedPagesCount, setLoadedPagesCount] = useState(1)
    const [scale, setScale] = useState(1)
    const [renderScale, setRenderScale] = useState(1)
    const [showSidebar, setShowSidebar] = useState(false)
    const [currentPage, setCurrentPage] = useState(0)
    const [pdfData, setPdfData] = useState(null)
    const [downloadProgress, setDownloadProgress] = useState(0)
    const [downloadError, setDownloadError] = useState(null)

    const virtuosoRef = useRef(null)
    const sidebarVirtuosoRef = useRef(null)
    const debounceRef = useRef(null)

    useEffect(() => {
        let active = true
        let localUrl = null

        setPdfData(null)
        setDownloadProgress(0)
        setDownloadError(null)

        const downloadFile = async () => {
            try {
                const response = await fetch(fileUrl, {
                    credentials: "include"
                })
                if (!response.ok) {
                    throw new Error(`Failed to download PDF: Server returned ${response.status}`)
                }

                const contentLength = response.headers.get("content-length")
                const total = contentLength ? parseInt(contentLength, 10) : 0

                const reader = response.body.getReader()
                let loaded = 0
                const chunks = []

                while (true) {
                    const { done, value } = await reader.read()
                    if (done) break

                    chunks.push(value)
                    loaded += value.length

                    if (total && active) {
                        setDownloadProgress(Math.round((loaded / total) * 100))
                    }
                }

                if (!active) return

                const allChunks = new Uint8Array(loaded)
                let position = 0
                for (const chunk of chunks) {
                    allChunks.set(chunk, position)
                    position += chunk.length
                }

                const blob = new Blob([allChunks], { type: "application/pdf" })
                localUrl = URL.createObjectURL(blob)
                setPdfData(localUrl)
            } catch (error) {
                console.error("Error downloading PDF:", error)
                if (active) {
                    setDownloadError(error.message || "Failed to download PDF")
                }
            }
        }

        downloadFile()

        return () => {
            active = false
            if (localUrl) {
                URL.revokeObjectURL(localUrl)
            }
        }
    }, [fileUrl])

    const onLoadSuccess = ({ numPages }) => {
        setNumPages(numPages)
        setLoadedPagesCount(Math.min(5, numPages))
    }

    const clampScale = (v) => Math.min(Math.max(v, 0.5), 2.5)

    const scheduleRender = (newScale) => {
        if (debounceRef.current) clearTimeout(debounceRef.current)
        debounceRef.current = setTimeout(() => {
            setRenderScale(newScale)
        }, 300)
    }

    const zoomIn = () => {
        setScale(prev => {
            const next = clampScale(prev + 0.2)
            scheduleRender(next)
            return next
        })
    }

    const zoomOut = () => {
        setScale(prev => {
            const next = clampScale(prev - 0.2)
            scheduleRender(next)
            return next
        })
    }

    const resetZoom = () => {
        setScale(1)
        scheduleRender(1)
    }

    const handlePageClick = (index) => {
        if (index >= loadedPagesCount) {
            setLoadedPagesCount(index + 1)
        }
        setCurrentPage(index)
        setTimeout(() => {
            virtuosoRef.current?.scrollToIndex({
                index,
                align: "start",
                behavior: "auto"
            })
        }, 80)
    }

    // sync sidebar scrolling with active page
    useEffect(() => {
        if (showSidebar && sidebarVirtuosoRef.current) {
            sidebarVirtuosoRef.current.scrollToIndex({
                index: currentPage,
                align: "center",
                behavior: "smooth"
            })
        }
    }, [currentPage, showSidebar])

    if (!pdfData) {
        return (

        disableAutoFetch: true,
        disableStream: false,
    }), [])

    const onLoadSuccess = ({ numPages }) => {
        setNumPages(numPages)
        setLoadedPagesCount(Math.min(5, numPages))

            setRenderScale(newScale)

        setScale(prev => {
            const next = clampScale(prev + 0.2)

                behavior: "smooth"

                    <span>Failed to load PDF</span>

            {showSidebar && numPages && (
                <div 
                    style={{
                        width: "200px",
                        minWidth: "200px",
                        height: "100%",
                        borderRight: "1px solid var(--secondary)",
                        backgroundColor: "var(--light)",
                        display: "flex",

//  helper function for socket notify all user that some change made
export const notifySharedUsers = async (itemId, event, data, emitToUser) => {
  let current = await Upload.findById(itemId).select("sharedWith parent owner")

  while (current) {
    // always notify owner
    if (current.owner) {
      emitToUser(current.owner.toString(), event, data)
    }

    // notify shared users if any
    if (current.sharedWith?.length > 0) {
      current.sharedWith.forEach(s => {
        emitToUser(s.userId.toString(), event, data)
      })
      break
    }

    if (!current.parent) break
    current = await Upload.findById(current.parent).select("sharedWith parent owner")
  }
}

//  gettig all files and folders for current user
export const getUserFiles = async (req, res) => {
    try {
        const parent =
            !req.query.parent || req.query.parent === "null"
                ? null
                : req.query.parent

        const userId = req.user._id

        // pagination
        const limit = parseInt(req.query.limit) || 50
        const skip = parseInt(req.query.skip) || 0

        // ROOT LEVEL
        // own items + shared items
        if (!parent) {

            // own items count
            const ownCount = await Upload.countDocuments({
                owner: userId,
                parent: null,
                isTrashed: { $ne: true },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            })

            // calculate split pagination
            const ownSkip = Math.min(skip, ownCount)
            const ownLimit = Math.min(limit, ownCount - ownSkip)

            const sharedSkip = Math.max(0, skip - ownCount)
            const sharedLimit = Math.max(0, limit - ownLimit)

            // own items
            const ownItems = await Upload.find({
                owner: userId,
                parent: null,
                isTrashed: { $ne: true },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            })
                .select(
                    "name type fileSize fileType createdAt parent color isShared owner storagePath sharedWith"
                )
                .populate("owner", "_id name profilePic")
                .populate("sharedWith.userId", "_id name")
                .sort({ type: -1, createdAt: -1 })
                .skip(ownSkip)
                .limit(ownLimit)
                .lean()

            // shared items
            const sharedItems = await Upload.find({
                "sharedWith.userId": userId,
                isTrashed: { $ne: true },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            })
                .select(
                    "name type fileSize fileType createdAt parent color isShared owner storagePath sharedWith"
                )
                .populate("owner", "_id name profilePic")
                .populate("sharedWith.userId", "_id name")
                .sort({ type: -1, createdAt: -1 })
                .skip(sharedSkip)
                .limit(sharedLimit)
                .lean()

            // total
            const sharedCount = await Upload.countDocuments({
                "sharedWith.userId": userId,
                isTrashed: { $ne: true },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            })

            const total = ownCount + sharedCount

            // only first request send all ids
            let allIds = []

            if (skip === 0) {
                const [ownIds, sharedIds] = await Promise.all([
                    Upload.find({
                        owner: userId,
                        parent: null,
                        isTrashed: { $ne: true },
                        $or: [
                            { type: "folder" },
                            { type: "file", uploadStatus: "completed" }
                        ]
                    })
                        .select("_id")
                        .lean(),

                    Upload.find({
                        "sharedWith.userId": userId,
                        isTrashed: { $ne: true },
                        $or: [
                            { type: "folder" },
                            { type: "file", uploadStatus: "completed" }
                        ]
                    })
                        .select("_id")
                        .lean()
                ])

                allIds = [
                    ...ownIds.map(i => i._id),
                    ...sharedIds.map(i => i._id)
                ]
            }

            // fix storage path
            const fixPath = (item) => ({
                ...item,
                storagePath: item.storagePath
                    ? item.storagePath.split("files")[1]?.replace(/\\/g, "/")
                    : null
            })

            // mark shared items
            const markedSharedItems = sharedItems.map(item => ({
                ...fixPath(item),
                isSharedWithMe: true,
                permission:
                    item.sharedWith.find(
                        s => s.userId?._id?.toString() === userId.toString()
                    )?.permission || null
            }))

            const combined = [...ownItems.map(fixPath), ...markedSharedItems];
            combined.sort((a, b) => (a.type === "folder" ? -1 : 1) - (b.type === "folder" ? -1 : 1) || new Date(b.createdAt) - new Date(a.createdAt));

            return res.status(200).json({
                success: true,
                items: combined,
                allIds,
                total,
                hasMore:
                    skip + (ownItems.length + sharedItems.length) < total
            })
        }

        // INSIDE FOLDER

        // permission check
        const permission = await getUserPermission(userId, parent)

        if (!permission) {
            return res.status(403).json({
                success: false,
                message: "Access denied"
            })
        }

        // folder items
        const items = await Upload.find({
            parent,
            isTrashed: { $ne: true },
            $or: [
                { type: "folder" },
                { type: "file", uploadStatus: "completed" }
            ]
        })
            .select(
                "name type fileSize fileType createdAt parent color isShared owner storagePath sharedWith"
            )
            .populate("owner", "_id name profilePic")
            .populate("sharedWith.userId", "_id name")
            .sort({ type: -1, createdAt: -1 })
            .skip(skip)
            .limit(limit)
            .lean()

        // total
        const total = await Upload.countDocuments({
            parent,
            isTrashed: { $ne: true },
            $or: [
                { type: "folder" },
                { type: "file", uploadStatus: "completed" }
            ]
        })

        // all ids only first request
        let allIds = []

        if (skip === 0) {
            const idDocs = await Upload.find({
                parent,
                isTrashed: { $ne: true },
                $or: [
                    { type: "folder" },
                    { type: "file", uploadStatus: "completed" }
                ]
            })
                .select("_id")
                .lean()

            allIds = idDocs.map(i => i._id)
        }

        // fix path
        const fixPath = (item) => ({
            ...item,
            storagePath: item.storagePath
                ? item.storagePath.split("files")[1]?.replace(/\\/g, "/")
                : null
        })

        const markedItems = items.map(item => ({
            ...fixPath(item),
            permission,
            isSharedWithMe: permission !== "owner"
        }))

        return res.status(200).json({
            success: true,
            items: markedItems,
            allIds,
            total,
            hasMore: skip + items.length < total
        })

    } catch (error) {
        return res.status(500).json({
            success: false,
            message: error.message
        })
    }
}



// get current folder when user double cliks on the folder
export const getFolderPath = async (req, res) => {
  try {
    const { id } = req.params;
    const userId = req.user._id

    const trail = [];
    let currentId = id;

    while (currentId) {
      const folder = await Upload.findOne({
        _id: currentId,
        type: "folder"
      }).select("_id name parent owner sharedWith").lean();

      if (!folder) break;

      const permission = await getUserPermission(userId, folder._id)
      if (!permission) break;

      trail.unshift({ id: folder._id, name: folder.name })
      currentId = folder.parent
    }

    if (!trail.length) {
      return res.status(404).json({ success: false, message: "Folder not found" });
    }

    res.json({ success: true, trail });

  } catch (error) {
    res.status(500).json({ success: false, message: error.message });
  }
}






//  rename file and folders
export const renameItem = async (req, res) => {
  try {
    const { id, newName } = req.body;

    if (!newName || !newName.trim()) {
      return res.status(400).json({ message: "Name is required" });
    }

    // here for shared folder or file check if owner or editor other wise permission denied
    const permission = await getUserPermission(req.user._id, id)
    if (!permission || !["owner", "editor"].includes(permission)) {
      return res.status(403).json({ success: false, message: "Access denied" })
    }

    //  first find the exact item here
    const item = await Upload.findOne({
      _id: id
    })

    if (!item) {
      return res.status(400).json({ success: false, message: "Not Found" })
    }

    //  prevent same name inside current folder
    const exists = await Upload.findOne({
      name: newName.trim(),
      parent: item.parent,
      owner: req.user._id,
      _id: { $ne: id },
      isTrashed: { $ne: true }
    })

    if (exists) {
      return res.status(400).json({ message: "Name already exists in this folder" });
    }

    //  rename it
    item.name = newName.trim()
    await item.save()

    // socket evetn to nofify other user here
    await notifySharedUsers(id, "item_renamed", { itemId: id, parentId: item.parent }, req.emitToUser)

    res.json({ success: true, item });

  } catch (error) {
    res.status(500).json({ message: error.message });
  }
}





//  For delete files and folders
export const deleteItem = async (req, res) => {
  try {
    const { id } = req.body;


    //  here for shared folder or file only owner can delete them
    const permission = await getUserPermission(req.user._id, id)
    if (permission !== "owner") {
      return res.status(403).json({ success: false, message: "Access denied" })
    }

    const item = await Upload.findOne({
      _id: id
    });

    if (!item) {
      return res.status(404).json({ message: "Not Found" });
    }


    // socket here for notify all users
    await notifySharedUsers(item.parent || id, "item_deleted", { itemId: id, parentId: item.parent }, req.emitToUser)


    // FILE check here for storage count before deleting from disk
    if (item.type === "file") {
      const count = await Upload.countDocuments({ storagePath: item.storagePath })
      if (count === 1 && item.storagePath && fs.existsSync(item.storagePath)) {
        fs.unlinkSync(item.storagePath)
      }
      await Upload.deleteOne({ _id: id });
      return res.status(200).json({ success: true, message: "Deleted" });
    }

    // if delete folder delete also nested children recursive delete
    const deleteRecursive = async (parentId) => {
      const children = await Upload.find({ parent: parentId });

      for (const child of children) {
        if (child.type === "folder") {
          await deleteRecursive(child._id);
        } else if (child.type === "file") {
          const count = await Upload.countDocuments({ storagePath: child.storagePath })
          if (count === 1 && child.storagePath && fs.existsSync(child.storagePath)) {
            fs.unlinkSync(child.storagePath)
          }
        }
        await Upload.deleteOne({ _id: child._id });
      }
    };

    await deleteRecursive(id);
    await Upload.deleteOne({ _id: id });

    return res.status(200).json({ success: true, message: "Deleted" });

  } catch (error) {
    res.status(500).json({ message: error.message });
  }
};





//  here user can change folder color like red green yellow
export const changeItemColor = async (req, res) => {
  try {
    const { ids, color } = req.body;

    //  validation - if no folder selection
    if (!ids || !Array.isArray(ids) || ids.length === 0) {
      return res.status(400).json({ success: false, message: "Please select a Folder" })
    }

    //  validation - if no color selected
    if (!color) {
      return res.status(400).json({ success: false, message: "Please select color" })
    }


    //  check here permission here only owner and editor have 
    for (const id of ids) {
      const permission = await getUserPermission(req.user._id, id)
      if (!permission || !["owner", "editor"].includes(permission)) {
        return res.status(403).json({ success: false, message: "Access denied" })
      }
    }

    // only for folder update color
    await Upload.updateMany({
      _id: { $in: ids },
      type: "folder"
    },
      {
        $set: { color }
      }
    )


    //  socket event for other uses
    for (const id of ids) {
      await notifySharedUsers(id, "item_color_changed", { itemId: id, color }, req.emitToUser)
    }


    return res.status(200).json({ success: true, message: "Color Changed" })

  } catch (error) {
    res.status(500).json({ success: false, message: error.message })
  }
}






//  movine folder or file here 
export const moveItem = async (req, res) => {
  try {
    const { itemId, destinationId } = req.body;


    // for shared folder and file allow owner and editor to move items
    const permission = await getUserPermission(req.user._id, itemId)
    if (permission !== "owner") {
      return res.status(403).json({ success: false, message: "Access denied" })
    }


    // fetch the item is being moved
    const item = await Upload.findOne({ _id: itemId })
    if (!item) {
      return res.status(404).json({ success: false, message: "Item not found" })
    }

    // if folder or file already in that location
    const destIdStr = destinationId ? destinationId.toString() : null
    const currentParentStr = item.parent ? item.parent.toString() : null
    if (currentParentStr === destIdStr) {
      return res.status(400).json({ success: false, message: "item is already in this folder" })
    }

    // can't move into it self
    if (destinationId && item._id.toString() === destinationId.toString()) {
      return res.status(400).json({ success: false, message: "Cannot move a folder into itself" });
    }


    //  validation destination folder is exist or not
    if (destinationId) {
      const destPermission = await getUserPermission(req.user._id, destinationId)
      if (!destPermission || !["owner", "editor"].includes(destPermission)) {
        return res.status(403).json({ success: false, message: "Access denied" })
      }

      const destination = await Upload.findOne({
        _id: destinationId,
        type: "folder"
      })
      if (!destination) {
        return res.status(404).json({ success: false, message: "Destination folder not found" });
      }
      if (destination.isTrashed) {
        return res.status(400).json({ success: false, message: "Cannot move items into a trashed folder" });
      }
    }

    //  folder can't move into its own subfolder 

    // ── 5. If moving a folder → check circular reference ───────────
    if (item.type === "folder" && destinationId) {
      let currentId = destinationId;
      while (currentId) {
        if (currentId.toString() === item._id.toString()) {
          return res.status(400).json({
            message: "Cannot move a folder into its own subfolder"
          });
        }
        const parent = await Upload.findOne({
          _id: currentId,
          owner: req.user._id
        }).select("parent");
        if (!parent) break;
        currentId = parent.parent;
      }
    }

    //  name conflict check 
    const conflict = await Upload.findOne({
      name: item.name,
      parent: destinationId || null,
      _id: { $ne: itemId },
      isTrashed: { $ne: true }
    })
    if (conflict) {
      return res.status(400).json({
        message: `A ${conflict.type} named "${item.name}" already exists in the destination`
      });
    }


    //  move item 
    const oldParent = item.parent
    item.parent = destinationId || null


    await item.save()

    //  socket evetn for notify all users
    await notifySharedUsers(itemId, "item_moved", { itemId, oldParent, newParent: destinationId || null }, req.emitToUser)

    res.json({ success: true, item });

  } catch (error) {
    res.status(500).json({ message: error.message });
  }
}





//  copy item here 
export const copyItem = async (req, res) => {
  try {
    const { itemId, destinationId } = req.body;


    //  for shared folder and file only owenr and editor
    const permission = await getUserPermission(req.user._id, itemId)
    if (!permission || permission === "viewer") {
      return res.status(403).json({ success: false, message: "Viewers cannot copy items" })
    }

    // ── 1. Fetch item being copied ──────────────────────────────────
    const item = await Upload.findOne({ _id: itemId });
    if (!item) {
      return res.status(404).json({ message: "Item not found" });
    }

    // ── 2. Validate destination exists ─────────────────────────────
    if (destinationId) {
      const destPermission = await getUserPermission(req.user._id, destinationId)
      if (!destPermission || !["owner", "editor"].includes(destPermission)) {
        return res.status(403).json({ success: false, message: "Access denied" })
      }

      const destination = await Upload.findOne({
        _id: destinationId,
        type: "folder"
      });
      if (!destination) {
        return res.status(404).json({ message: "Destination folder not found" });
      }
      if (destination.isTrashed) {
        return res.status(400).json({ success: false, message: "Cannot copy items into a trashed folder" });
      }
    }

    // ── 3. Name conflict check ──────────────────────────────────────
    let copyName = item.name;
    const conflict = await Upload.findOne({
      name: copyName,
      parent: destinationId || null,
      isTrashed: { $ne: true }
    });
    if (conflict) {
      // append - Copy to name
      const ext = copyName.includes(".")
        ? "." + copyName.split(".").pop()
        : "";
      const base = copyName.includes(".")
        ? copyName.substring(0, copyName.lastIndexOf("."))
        : copyName;
      copyName = ext ? `${base} - Copy${ext}` : `${base} - Copy`;
    }

    // ── 4. Recursive copy function ──────────────────────────────────
    const copyRecursive = async (sourceItem, newParentId, newName) => {
      const newDoc = await Upload.create({
        name: newName || sourceItem.name,
        type: sourceItem.type,
        parent: newParentId,
        owner: req.user._id,
        fingerprint: sourceItem.fingerprint,
        fileSize: sourceItem.fileSize,
        fileType: sourceItem.fileType,
        storagePath: sourceItem.storagePath, // same path on disk
        uploadStatus: sourceItem.type === "file" ? "completed" : null,
        color: sourceItem.color,
        uploadId: null,
        totalChunks: sourceItem.totalChunks,
        refCount: 1,
        lastActivity: null,
      });

      // if folder → copy all children recursively
      if (sourceItem.type === "folder") {
        const children = await Upload.find({
          parent: sourceItem._id
        });

        for (const child of children) {
          await copyRecursive(child, newDoc._id, null)
        }
      }

      return newDoc;
    }

    // ── 5. Do the copy ──────────────────────────────────────────────
    const newItem = await copyRecursive(item, destinationId || null, copyName);


    //  socket evetn here for notify all users here
    if (destinationId) {
      await notifySharedUsers(destinationId, "item_copied", { parentId: destinationId }, req.emitToUser)

    }

    res.json({ success: true, item: newItem });

  } catch (error) {
    res.status(500).json({ message: error.message });
  }
};






//  for creating new empty folde here
export const createFolder = async (req, res) => {
  try {
    const { name, parentId } = req.body;
    const userId = req.user._id

    if (!name || !name.trim()) {
      return res.status(400).json({ message: "Folder name is required" })
    }

    // check permissions and trash status for the parent folder
    if (parentId) {
      const parentFolder = await Upload.findById(parentId).select("isTrashed");
      if (parentFolder && parentFolder.isTrashed) {
        return res.status(400).json({
          success: false,
          message: "Cannot create folder in a trashed folder"
        });
      }

      const permission = await getUserPermission(userId, parentId)
      if (!permission || !["owner", "editor"].includes(permission)) {
        return res.status(403).json({
          success: false,
          message: "You don't have permission to create folders here"
        })
      }
    }


    // prevent duplicate fodler creation in same parent here
    const exists = await Upload.findOne({
      name,
      parent: parentId || null,
      owner: userId,
      type: "folder",
      isTrashed: { $ne: true }
    })

    if (exists) {
      return res.status(400).json({ message: "Folder already exists" })
    }


    // create folder
    const folder = await Upload.create({
      name,
      parent: parentId || null,
      owner: userId,
      type: "folder"
    })

    if (parentId) {
      await notifySharedUsers(parentId, "item_uploaded", { folderId: String(parentId) }, req.emitToUser)
    }

    res.status(201).json({ folder })

  } catch (error) {
    res.status(500).json({ message: "Failed to create folder" })
  }
}
